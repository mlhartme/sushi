39      @Test
40      public void empty() {
41          Parser parser;
42          Empty empty;
43  
44          parser = Parser.create(METADATA, Empty.class);
45          empty = new Empty();
46          parser.run(empty);
47          try {
48              parser.run(empty, "-a", "10");
49              fail();
50          } catch (ArgumentException e) {
51              assertTrue(e.getMessage().contains("unknown option"));
52          }
53          try {
54              parser.run(empty, "a");
55              fail();
56          } catch (ArgumentException e) {
57              assertTrue(e.getMessage(), e.getMessage().contains("unknown value"));
58          }
59          try {
60              parser.run(empty, "-");
61              fail();
62          } catch (ArgumentException e) {
63              assertTrue(e.getMessage(), e.getMessage().contains("unknown value"));
64          }
65      }

67      @Test
68      public void values() throws Exception {
69          Parser parser;
70          Values values;
71  
72          parser = Parser.create(METADATA, Values.class);
73  
74          values = new Values();
75          try {
76              parser.run(values);
77              fail();
78          } catch (ArgumentException e) {
79              assertTrue(e.getMessage().contains("missing"));
80          }
81  
82          values = new Values();
83          try {
84              parser.run(values, "first");
85              fail();
86          } catch (ArgumentException e) {
87              assertTrue(e.getMessage().contains("missing"));
88          }
89  
90          values = new Values();
91          parser.run(values, "first", "second");
92          assertEquals("first", values.first.getName());
93          assertEquals("second", values.second);
94          assertEquals(0, values.remaining.size());
95  
96          values = new Values();
97          parser.run(values, "first", "second", "third", "forth");
98          assertEquals("first", values.first.getName());
99          assertEquals("second", values.second);
100         assertEquals(2, values.remaining.size());
101         assertEquals(WORLD.file("third"), values.remaining.get(0));
102         assertEquals(WORLD.file("forth"), values.remaining.get(1));
103 
104         values = new Values();
105         parser.run(values, "first", "second", "-");
106         assertEquals("first", values.first.getName());
107         assertEquals("second", values.second);
108         assertEquals(1, values.remaining.size());
109         assertEquals(WORLD.file("-"), values.remaining.get(0));
110     }

112     @Test
113     public void options() {
114         Parser parser;
115         Options options;
116 
117         parser = Parser.create(METADATA, Options.class);
118         options = new Options();
119         parser.run(options);
120         assertEquals(0, options.first);
121         assertNull(options.second);
122         assertFalse(options.third);
123 
124         options = new Options();
125         assertSame(options, parser.run(options, "-first", "1"));
126         assertEquals(1, options.first);
127         assertNull(options.second);
128         assertFalse(options.third);
129 
130         options = new Options();
131         try {
132             assertSame(options, parser.run(options, "-first"));
133             fail();
134         } catch (ArgumentException e) {
135             assertTrue(e.getMessage(), e.getMessage().contains("missing value"));
136         }
137 
138         options = new Options();
139         parser.run(options, "-second", "foo");
140         assertEquals(0, options.first);
141         assertEquals("foo", options.second);
142         assertFalse(options.third);
143 
144         options = new Options();
145         parser.run(options, "-third");
146         assertEquals(0, options.first);
147         assertNull(options.second);
148         assertTrue(options.third);
149 
150         options = new Options();
151         parser.run(options, "-third", "-first", "-1", "-second", "bar");
152         assertEquals(-1, options.first);
153         assertEquals("bar", options.second);
154         assertTrue(options.third);
155 
156         options = new Options();
157         try {
158             parser.run(options, "-first");
159             fail();
160         } catch (ArgumentException e) {
161             assertTrue(e.getMessage().contains("missing"));
162         }
163 
164         options = new Options();
165         try {
166             parser.run(options, "-first", "a");
167             fail();
168         } catch (ArgumentException e) {
169             assertTrue(e.getMessage(), e.getMessage().contains("expected integer"));
170         }
171     }

173     @Test
174     public void children() {
175         Parser parser;
176         Children children;
177         ChildObject child;
178 
179         children = new Children();
180         parser = Parser.create(METADATA, Children.class);
181         child = (ChildObject) parser.run(children, "a");
182         assertEquals("a", child.name);
183         assertEquals(0, child.remaining.size());
184 
185         children = new Children();
186         parser = Parser.create(METADATA, Children.class);
187         child = (ChildObject) parser.run(children, "b", "1", "2");
188         assertEquals("b", child.name);
189         assertEquals(Arrays.asList("1", "2"), child.remaining);
190     }

192     @Test
193     public void childAndValue() {
194         Parser parser;
195         ChildAndValue instance;
196         ChildObject child;
197 
198         parser = Parser.create(METADATA, ChildAndValue.class);
199         instance = new ChildAndValue();
200         child = (ChildObject) parser.run(instance, "a");
201         assertEquals("a", child.name);
202         assertNull(instance.b);
203 
204         instance = new ChildAndValue();
205         assertSame(instance, parser.run(instance, "b"));
206         assertEquals("b", instance.b);
207    }
