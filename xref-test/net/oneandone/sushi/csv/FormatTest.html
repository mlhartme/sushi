104         cell("' 1 '", "' 1 '\n", " 1 ");
105         cell("1 | 2", "'1 | 2'\n", "1 ", " 2");

107     }

109     // test lines where each cell has exacly one value
110     private void values(String line, String lineExpected, String ... cellsExpected) throws CsvLineException {
111         Line parsed;
112         List<String> cell;
113         
114         parsed = format.read(line);
115         assertEquals(cellsExpected.length, parsed.size());
116         for (int i = 0; i < cellsExpected.length; i++) {
117             cell = parsed.get(i);
118             assertEquals(1, cell.size());
119             assertEquals(cellsExpected[i], cell.get(0));
120         }
121         assertEquals(lineExpected, format.write(parsed));
122     }
123 
124     // test lines with exactly one cell
125     private void cell(String line, String lineExpected, String ... valuesExpected) throws CsvLineException {
126         Line parsed;
127         List<String> lst;
128         
129         parsed = format.read(line);
130         assertEquals(1, parsed.size());
131         lst = valuesExpected == null ? null : Arrays.asList(valuesExpected);
132         assertEquals(lst, parsed.get(0));
133         assertEquals(lineExpected, format.write(parsed));
134     }
