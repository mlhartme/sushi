37  	public CopyDiffTest() throws IOException {
38  		world = new World();
39  		variables = new HashMap<>();
40          copy = new Copy(world.getTemp().createTempDirectory(),
41                  world.filter().includeAll(), Filter.NOTHING, world.getTemp().getRoot().getFilesystem().getFeatures().modes,
42                  variables, Substitution.path(), Substitution.ant(), Copy.DEFAULT_CONTEXT_DELIMITER, Copy.DEFAULT_CALL_PREFIX);

45      @Test
46      public void mode() throws Exception {
47          Node destdir;
48          Node file;
49          
50          if (!copy.getSourceDir().getRoot().getFilesystem().getFeatures().modes) {
51          	return;
52          }
53          destdir = copy.getSourceDir().getWorld().getTemp().createTempDirectory();
54  
55          file = copy.getSourceDir().join("file");
56          file.writeString("foo");
57          file.setPermissions("rwx------");
58          copy.directory(destdir);
59          assertEquals("rwx------", destdir.join("file").getPermissions());
60  
61          file.setPermissions("rw-r-xr-x");
62          assertEquals(l("m file"), brief(destdir));
63          copy.directory(destdir);
64          assertEquals("rw-r-xr-x", destdir.join("file").getPermissions());
65      }

67  	@Test
68  	public void diff() throws Exception {
69  		Node left;
70  		Node right;
71  		
72  		left = world.getTemp().createTempDirectory();
73  		right = world.getTemp().createTempDirectory();
74  		left.join("left").writeString("1");
75  		right.join("right").writeString("2");
76  		assertEquals(l("R left", "A right"), left.diffDirectory(right, true));
77  		assertEquals(l("A right"), new Diff(true).directory(left, right, "right"));
78  	}

80  	@Test
81      public void template() throws Exception {
82          Node destdir;
83          String brief;
84          String normal;
85          
86          destdir = copy.getSourceDir().getWorld().getTemp().createTempDirectory();
87          variables.put("home", "mhm");
88          variables.put("machine", "walter");
89          
90          assertEquals("", brief(destdir));
91          assertEquals("", diff(destdir));
92  
93          copy.getSourceDir().join("file").writeLines("home: ${home}", "machine: ${machine}");
94          assertEquals(l("A file"), brief(destdir));
95          assertEquals(l("### file",
96          		"+ home: mhm",
97                  "+ machine: walter"), diff(destdir));
98          copy.directory(destdir);
99          assertEquals("", brief(destdir));
100         
101         copy.getSourceDir().join("folder").mkdir();
102         assertEquals(l("A folder"), brief(destdir));
103         copy.directory(destdir);
104         assertEquals("", brief(destdir));
105         
106         copy.getSourceDir().join("superdir/subdir").mkdirs();
107         assertEquals(l("A superdir", "A superdir/subdir"), brief(destdir));
108         copy.directory(destdir);
109         assertEquals("", brief(destdir));
110 
111         copy.getSourceDir().join("folder/file").writeLines("home: ${home}", "machine: ${machine}");
112         assertEquals(l("A folder/file"), brief(destdir));
113         copy.directory(destdir);
114         assertEquals("", brief(destdir));
115         
116         variables.put("machine", "fritz");
117         brief = brief(destdir);
118         assertTrue(brief, brief.equals(l("M folder/file", "M file")) || brief.equals(l("M file", "M folder/file")));
119         normal = diff(destdir);
120         assertTrue(normal,
121                 (l("### folder/file",
122                 "-machine: walter",
123                 "+machine: fritz",
124                 "### file",
125                 "-machine: walter",
126                 "+machine: fritz")).equals(normal) ||
127                 (l("### file",
128                 "-machine: walter",
129                 "+machine: fritz",
130                 "### folder/file",
131                 "-machine: walter",
132                 "+machine: fritz")).equals(normal)
133                 );
134         copy.directory(destdir);
135         assertEquals("", brief(destdir));
136         assertEquals("", diff(destdir));
137     }

171     private String diff(Node destdir) throws IOException {
172         return doDiff(destdir, false);
173     }
174 
175     private String brief(Node destdir) throws IOException {
176         return doDiff(destdir, true);
177     }
178     
179     private String doDiff(Node destdir, boolean brief) throws IOException {
180         Node tmp = world.getTemp().createTempDirectory();
181         copy.directory(tmp);
182         return destdir.diffDirectory(tmp, brief);
183     }

185     public static class CopyExt extends Copy {
186         public String called = null;
187         
188         public CopyExt(Node srcdir, Map<String, String> variables) {
189             super(srcdir, srcdir.getWorld().filter().includeAll(), false, variables, Copy.DEFAULT_SUBST, Copy.DEFAULT_SUBST, '-', '@');
190         }
191         
192         public List<Map<String, String>> contextN(Map<String, String> parent) {
193             return ctx(parent, "n");
194         }
195 
196         public List<Map<String, String>> contextMoreNumbers(Map<String, String> parent) {
197             return ctx(parent, "m");
198         }
199 
200         public void callTestDir(Node node, Map<String, String> context) {
201             called = node.getName();
202         }
203 
204         public String callTestFile(Map<String, String> context) {
205             return context.get("var");
206         }
207         
208         private List<Map<String, String>> ctx(Map<String, String> parent, String name) {
209             List<Map<String, String>> result;
210             
211             result = new ArrayList<Map<String, String>>();
212             result.add(map(parent, name, 1));
213             result.add(map(parent, name, 2));
214             return result;
215         }
216 
217         private static Map<String, String> map(Map<String, String> parent, String name, int n) {
218             Map<String, String> result;
219             
220             result = new HashMap<String, String>(parent);
221             result.put(name, Integer.toString(n));
222             return result;
223         }
224     }
225     
226     private static String l(String ... lines) {
227     	return OS.CURRENT.lines(lines);
228     }
