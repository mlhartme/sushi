16  package net.oneandone.sushi.fs;
17  
18  import net.oneandone.sushi.io.Buffer;
19  import net.oneandone.sushi.io.OS;
20  import org.junit.After;
21  import org.junit.Before;
22  import org.junit.Test;
23  
24  import java.io.IOException;
25  import java.net.URI;
26  
27  import static org.junit.Assert.assertEquals;
28  
29  public abstract class NodeReadOnlyTest<T extends Node> {
30      protected static final World WORLD = new World(OS.CURRENT, new Settings(), new Buffer(), "**/.svn/**/*").addStandardFilesystems(true);
31  
32      /** creates a new empty directory */
33      protected abstract T createWork() throws IOException;
34  
35      protected T work;
36  
37      @Before
38      public void setUp() throws Exception {
39          work = createWork();
40          validateDeallocation();
41      }
42  
43      @After
44      public abstract void validateDeallocation() throws Exception;
45  
46      @Test
47      public void uri() throws Exception {
48          URI uri;
49          Node again;
50          Filesystem fs;
51  
52          fs = work.getRoot().getFilesystem();
53          uri = work.getURI();
54          assertEquals(uri, work.getWorld().node(fs.getScheme() + ":"
55                  + work.getRoot().getId() + Node.encodePath(work.getPath())).getURI());
56          again = WORLD.node(uri);
57          assertEquals(work, again);
58          assertEquals(uri, again.getURI());
59      }
60  }
