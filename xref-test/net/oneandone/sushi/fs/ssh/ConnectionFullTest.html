35      private static final World WORLD = new World();

37      public static SshRoot open() throws JSchException, IOException {
38          return WORLD.getFilesystem("ssh", SshFilesystem.class).root(host(), user());
39      }
40  
41      public static String host() throws JSchException, IOException {
42          String host;
43          InetAddress addr;
44  
45          host = TestProperties.get("ssh.host");
46          if (host == null) {
47              try {
48                  addr = InetAddress.getLocalHost();
49                  host = addr.getHostName();
50              } catch (UnknownHostException e) {
51                  host = "localhost";
52              }
53          }
54          return host;
55      }
56  
57      public static String user() throws JSchException, IOException {
58          String user;
59  
60          user = TestProperties.get("ssh.user");
61          if (user.isEmpty()) {
62              user = System.getProperty("user.name");
63          }
64          return user;
65      }

82      @Test
83      public void normal() throws Exception {
84          assertEquals("\r\n", root.exec("echo"));
85          assertEquals("hello\r\n", root.exec("echo", "hello"));
86          assertEquals("again\r\n", root.exec("echo", "again"));
87          assertEquals("err\r\n", root.exec("echo", "err", "1>&2"));
88          try {
89              root.exec("commandnotfound");
90              fail();
91          } catch (ExitCode e) {
92              assertTrue(e.output.contains("commandnotfound"));
93          }
94  
95          assertEquals("alive\r\n", root.exec("echo", "alive"));
96      }

124     @Test
125     public void shell() throws Exception {
126         root.exec("echo", "-e", "\\003320l");
127 
128         assertEquals("", root.exec("exit", "0", "||", "echo", "dontprintthis"));
129         assertEquals("a\r\nb\r\n", root.exec("echo", "a", "&&", "echo", "b"));
130         assertEquals("a\r\n", root.exec("echo", "a", "||", "echo", "b"));
131         assertEquals("file\r\n", root.exec(
132                 "if", "test", "-a", "/etc/profile;",
133                 "then", "echo", "file;",
134                 "else", "echo", "nofile;", "fi"));
135         assertEquals("nofile\r\n", root.exec(
136                 "if", "test", "-a", "nosuchfile;",
137                 "then", "echo", "file;",
138                 "else", "echo", "nofile;", "fi"));
139     }

171         String tmp = "/tmp/cancel-sushi";
172         String msg;
173 
174         root.exec("rm", "-f", tmp);
175         root.start(true, "sleep", "2", "&&", "echo", "hi", ">" + tmp);
176         Thread.sleep(500);
177         tearDown();
178         Thread.sleep(3000);
179         setUp();

181             root.exec("LANG=us", "ls", tmp);

240     private World identity(String name, String passphrase) throws Exception {
241         World world;
242         SshFilesystem fs;
243 
244         world = new World();
245         fs = world.getFilesystem("ssh", SshFilesystem.class);
246         fs.addIdentity(world.guessProjectHome(getClass()).join("src/test/ssh/" + name + "/id_rsa"), passphrase);
247         return world;
248     }

250     @Test
251     public void multiplIdentities() throws JSchException, IOException {
252         World world;
253         SshFilesystem fs;
254 
255         world = new World();
256         fs = world.getFilesystem("ssh", SshFilesystem.class);
257         fs.addDefaultIdentityOpt();
258         fs.addIdentity(world.guessProjectHome(getClass()).join("src/test/ssh/without-passphrase/id_rsa"), null);
259         fs.addIdentity(world.guessProjectHome(getClass()).join("src/test/ssh/with-passphrase/id_rsa"), "passphrase");
260         fs.root(host(), user()).exec("ls");
261     }
