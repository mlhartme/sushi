40      @Test
41      public void junit() throws Exception {
42          FileNode jar;
43          String rootPath;
44          String locator;
45          ZipNode assrt;
46          ZipNode junit;
47          Node root;
48          List<? extends Node> list;
49          List<? extends Node> tree;
50  
51          jar = world.locateClasspathItem(Assert.class);
52          rootPath = jar.getURI().toString() + "!/org/junit/Assert.class";
53          locator = "jar:" + rootPath;
54          assrt = (ZipNode) world.node(locator);
55          assertEquals(locator, assrt.getURI().toString());
56          assertEquals("org/junit/Assert.class", assrt.getPath());
57          assertTrue(assrt.exists());
58          assertTrue(assrt.isFile());
59          assertTrue(assrt.readString().length() > 100);
60          junit = assrt.getParent();
61          assertEquals("org/junit", junit.getPath());
62          assertTrue(junit.isDirectory());
63          list = junit.list();
64          assertTrue(list.size() > 10);
65          assertTrue(list.contains(assrt));
66          assertFalse(list.contains(list));
67          assertEquals(1, junit.getParent().list().size());
68          root = junit.getParent().getParent();
69          assertEquals("", root.getPath());
70          assertTrue(root.isDirectory());
71          assertTrue(root.exists());
72          tree = junit.find("**/*");
73          assertTrue(tree.size() > list.size());
74          assertTrue(tree.contains(assrt));
75          assertFalse(tree.contains(list));
76          assertTrue(tree.containsAll(list));
77          assrt = junit.join("Assert.class");
78          assertTrue(assrt.exists());
79          assertTrue(assrt.isFile());
80      }

82      @Test
83      public void type() throws Exception {
84          FileNode jar;
85          ZipNode root;
86          int count;
87  
88          jar = world.locateClasspathItem(Assert.class);
89          root = jar.openZip();
90          count = 0;
91          for (Node node : root.find("**/*")) {
92              count++;
93              while (true) {
94                  node = node.getParent();
95                  if (node == null) {
96                      break;
97                  }
98                  assertTrue(node.isDirectory());
99                  assertFalse(node.isFile());
100             }
101         }
102         assertEquals(266, count);
103     }

132     private void checkSpecialPath(String dir, String name) throws IOException {
133         final String clazz = "org/junit/Assert.class";
134         FileNode jar;
135         Node temp;
136         Node copy;
137         ZipNode zip;
138 
139         jar = world.locateClasspathItem(Assert.class);
140         temp = world.getTemp().createTempDirectory();
141         copy = temp.join(dir).mkdir().join(name);
142         jar.copyFile(copy);
143         zip = ((FileNode) copy).openZip();
144         assertEquals(1, zip.find(clazz).size());
145         assertNotNull(world.validNode("zip:" + copy.getURI() + "!/" + clazz).readBytes());
146         assertNotNull(world.validNode("jar:" + copy.getURI() + "!/" + clazz).readBytes());
147         temp.deleteTree();
148     }
