30  public class ComplexTypeTest {
31      private static final Schema METADATA = new AnnotationSchema();
32  
33      @Test
34      public void normal() {
35          ComplexType type;
36          Item item;
37          Car demo;
38  
39          type = (ComplexType) METADATA.type(Car.class);
40          assertEquals("car", type.getName());
41  
42          demo = (Car) type.newInstance();
43  
44          item = type.lookup("name");
45          assertEquals("name", item.getName());
46          assertEquals(String.class, item.getType().getType());
47          assertEquals("", item.getOne(demo));
48          item.setOne(demo, "foo");
49          assertEquals("foo", item.getOne(demo));
50          assertEquals(1, item.get(demo).size());
51  
52          item = type.lookup("seats");
53          assertEquals("seats", item.getName());
54          assertEquals(Integer.class, item.getType().getType());
55          item.setOne(demo, 7);
56          assertEquals(7, item.getOne(demo));
57      }
58  
59      @Test
60      public void sequence() {
61          ComplexType type;
62          Item item;
63          Vendor demo;
64  
65          type = (ComplexType) METADATA.type(Vendor.class);
66          assertEquals("vendor", type.getName());
67  
68          demo = (Vendor) type.newInstance();
69  
70          item = type.lookup("car");
71          assertEquals("car", item.getName());
72          assertEquals(METADATA.type(Car.class), item.getType());
73  
74          assertEquals(0, item.get(demo).size());
75      }
76  
77      @Test
78      public void clon() {
79          Car orig;
80          Car clone;
81  
82          orig = new Car("foo", Kind.SPORTS, 3, new Engine(true, 11), new Radio());
83          clone = METADATA.instance(orig).clone().get();
84          assertTrue(orig != clone);
85          assertEquals(3, clone.getSeats());
86          assertEquals("foo", clone.getName());
87          assertEquals(true, clone.getEngine().getTurbo());
88          assertEquals(11, clone.getEngine().getPs());
89          assertNotNull(clone.getRadio());
90      }
