16  package net.oneandone.sushi.rss;
17  
18  import junit.framework.TestCase;
19  import net.oneandone.sushi.xml.Xml;
20  import net.oneandone.sushi.xml.XmlException;
21  
22  import java.util.Date;
23  
24  public class FeedTest extends TestCase {
25      private static final Xml XML = new Xml();
26      
27      public void testEmpty() throws XmlException {
28          Feed feed;
29          
30          feed = new Feed();
31          feed = Feed.fromXml(XML.getSelector(), feed.toXml(XML.getBuilder()));
32          assertEquals(0, feed.channels().size());
33      }
34  
35      public void testNormal() throws XmlException {
36          Feed feed;
37          Channel channel;
38          Item item;
39          Date date = new Date();
40          
41          feed = new Feed();
42          channel = new Channel();
43          channel.items().add(new Item());
44          channel.items().add(new Item());
45          feed.channels().add(channel);
46          
47          channel = new Channel("t", "l", "d");
48          channel.items().add(new Item("t", "l", "d", "a", "g", date));
49          feed.channels().add(channel);
50          
51          feed = Feed.fromXml(XML.getSelector(), feed.toXml(XML.getBuilder()));
52          assertEquals(2, feed.channels().size());
53          assertEquals(2, feed.channels().get(0).items().size());
54  
55          channel = feed.channels().get(1);
56          assertEquals("t", channel.getTitle());
57          assertEquals("l", channel.getLink());
58          assertEquals("d", channel.getDescription());
59          assertEquals(1, channel.items().size());
60          
61          item = channel.items().get(0);
62          assertEquals("t", item.getTitle());
63          assertEquals("l", item.getLink());
64          assertEquals("d", item.getDescription());
65          assertEquals("a", item.getAuthor());
66          assertEquals("g", item.getGuid());
67          // dates do not equal because milli seconds get lost:
68          assertEquals(date.toString(), item.getPubDate().toString());
69      }
70  }
