16  package net.oneandone.sushi.util;
17  
18  import org.junit.Test;
19  
20  import static org.junit.Assert.assertEquals;
21  import static org.junit.Assert.assertSame;
22  
23  public class StringsTest {
24      @Test
25      public void indent() {
26          assertEquals("  a", Strings.indent("a", "  "));
27          assertEquals("  a\n   b", Strings.indent("a\n b", "  "));
28      }
29  
30      @Test
31      public void remove() {
32          final String STARTX = "startx";
33  
34          assertEquals("start", Strings.removeRight(STARTX, "x"));
35          assertEquals("start", Strings.removeRightOpt(STARTX, "x"));
36          assertEquals(STARTX, Strings.removeRightOpt(STARTX, "y"));
37          assertEquals("tartx", Strings.removeLeft(STARTX, "s"));
38          assertEquals("tartx", Strings.removeLeftOpt(STARTX, "s"));
39          assertSame(STARTX, Strings.removeLeftOpt(STARTX, "t"));
40      }
41  
42      @Test
43      public void pad() {
44          assertEquals("a", Strings.padLeft("a", 0));
45          assertEquals("a", Strings.padLeft("a", 1));
46          assertEquals(" a", Strings.padLeft("a", 2));
47          assertEquals("b", Strings.padRight("b", 0));
48          assertEquals("b", Strings.padRight("b", 1));
49          assertEquals("b ", Strings.padRight("b", 2));
50      }
51  
52      //--
53  
54      @Test
55      public void capitalize() {
56          assertEquals("", Strings.capitalize(""));
57          assertEquals("C", Strings.capitalize("c"));
58          assertEquals("Cap", Strings.capitalize("Cap"));
59          assertEquals("CaP", Strings.capitalize("caP"));
60          assertEquals("1", Strings.capitalize("1"));
61      }
62  
63      @Test
64      public void decapitalize() {
65          assertEquals("", Strings.decapitalize(""));
66          assertEquals("c", Strings.decapitalize("c"));
67          assertEquals("c", Strings.decapitalize("C"));
68          assertEquals("caP", Strings.decapitalize("CaP"));
69          assertEquals("1", Strings.decapitalize("1"));
70      }
71  
72      @Test
73      public void excape() {
74          assertEquals("", Strings.escape(""));
75          assertEquals("Hello, world", Strings.escape("Hello, world"));
76          assertEquals("a\\nb", Strings.escape("a\nb"));
77          assertEquals("\\\\\\n\\r\\t", Strings.escape("\\\n\r\t"));
78      }
79  
80      //---
81  
82      @Test
83      public void blockSimple() {
84          assertEquals("", Strings.block("", "", 0, ""));
85          assertEquals("foo", Strings.block("", "foo", 10, ""));
86          assertEquals("foo bar", Strings.block("", "foo bar", 10, ""));
87          assertEquals("barfoo", Strings.block("bar", "foo ", 10, ""));
88          assertEquals("foobar", Strings.block("", "foo", 10, "bar"));
89      }
90  
91      @Test
92      public void blockNormalize() {
93          assertEquals("foo", Strings.block("", " foo", 10, ""));
94          assertEquals("foo", Strings.block("", "foo ", 10, ""));
95          assertEquals("foo", Strings.block("", "foo\n", 10, ""));
96          assertEquals("foo bar", Strings.block("", "foo  bar", 10, ""));
97      }
98      @Test
99      public void blockBreak() {
100         assertEquals("foo-bar-", Strings.block("", "foo bar", 3, "-"));
101         assertEquals("foo-bar-", Strings.block("", "foo bar", 4, "-"));
102         assertEquals("foo-bar-", Strings.block("", "foo bar", 6, "-"));
103         assertEquals("foo bar-", Strings.block("", "foo bar", 7, "-"));
104         assertEquals("foo bar-", Strings.block("", "foo  bar", 7, "-"));
105     }
106 
107     @Test
108     public void blockToSmall() {
109         assertEquals("foo", Strings.block("", "foo", 0, ""));
110         assertEquals("foo-bar-", Strings.block("", "foo bar", 0, "-"));
111     }
112 }
