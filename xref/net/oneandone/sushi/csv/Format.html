16  package net.oneandone.sushi.csv;
17  
18  import java.io.IOException;
19  import java.io.StringWriter;
20  import java.io.Writer;
21  import java.util.List;
22  
23  /**
24   * Parser and Serializer for csv file lines.  
25   * 
26   * Note:
27   * o whitespace is always significant because it's never stripped by OOCalc 
28   *   (except for digit-only values).
29   * o comments are not supported because OOCalc would consider them as strings.
30   * o a line has at least one cell
31   * 
32   * You have to implement some pre- or post-processing if you want comments of whitespace handling.
33   *
34   * <pre>
35   * Grammar
36   * 
37   * line    = cell (SEP cell)* newline ;
38   * cell    = values 
39   *         | QUOTE values QUOTE ;
40   * values  = value (VSEP value)* 
41   *         | EMPTY
42   *         | NULL ;
43   * value   = char* ;
44   * char    = QUOTE QUOTE
45   *         | ESC CH
46   *         | SEP                               # aber nur wenn drumrum quotes stehen
47   *         | CH \ QUOTE \ ESC \ SEP \ VSEP  ;
48   * newline = CR | LF | CR LF ;
49   * </pre>
50   *
51   * Terminals as configured for the format:
52   *   QUOTE, SEP, VSEP, NEWLINE, COMMENT, NULL, EMPTY, ESC
53   *   
54   * Other Terminals:
55   *   CH: arbitrary unicode character
56   */
57  public class Format {
58      public final boolean merged;
59  
60      /** between cells */
61      private final char separator;
62      
63      /** between values */
64      private final char valueSeparator;
65  
66      private final String empty;
67      
68      private final String nul;
69      
70      /** escape special characters */
71      private final char escape;
72      
73      /** to quote cells */
74      private final char quote;
75  
76      private final String unquoted;
77      
78      public Format() {
79          this(false);
80      }
81      
82      public Format(boolean merged) {
83          this(merged, '\\', '"');
84      }
85      public Format(boolean merged, char escape, char quote) {
86          this(merged, "\n", ';', '|', "EMPTY", "NULL", escape, quote, "01234567890");
87      }
88      
89      public Format(boolean merged, String newline, char separator, char valueSeparator, 
90              String empty, String nul, char escape, char quote, String unquoted) {
91          this.merged = merged;
92          this.separator = separator;
93          this.valueSeparator = valueSeparator;
94          this.empty = "EMPTY";
95          this.nul = "NULL";
96          this.escape = escape;
97          this.quote = quote;
98          this.unquoted = unquoted;
99      }
100     
101     public Line read(String line) throws CsvLineException {
102         Line result;
103         Source src;
104         int c;
105         
106         result = new Line();
107         src = new Source(line);
108         while (true) {
109             cell(src, result);
110             c = src.peek();
111             if (c == Source.END) {
112                 return result;
113             }
114             if (c != separator) {
115                 throw new CsvLineException("separator expected");
116             }
117             src.eat();
118         }
119     }
120 
121     private void cell(Source src, Line result) throws CsvLineException {
122         boolean quoted;
123         
124         quoted = src.peek() == quote;
125         if (quoted) {
126             src.eat();
127         }
128         values(src, quoted, result);
129         if (quoted) {
130             if (src.peek() != quote) {
131                 throw new CsvLineException("quote not closed");
132             }
133             src.eat();
134         }
135     }
136 
137     private void values(Source src, boolean quoted, Line result) throws CsvLineException {
138         List<String> values;
139 
140         if (src.eat(nul, quoted ? quote : separator)) {
141             result.addNull();
142             return;
143         }
144         values = result.add();
145         if (src.eat(empty, quoted ? quote : separator)) {
146             return;
147         }
148         while (true) {
149             value(src, quoted, values);
150             if (src.peek() != valueSeparator) {
151                 return;
152             }
153             src.eat();
154         }
155     }
156 
157     private void value(Source src, boolean quoted, List<String> values) throws CsvLineException { 
158         StringBuilder builder;
159         int c;
160         
161         builder = new StringBuilder();
162         while (true) {
163             c = src.peek();
164             if (c == -1) {
165                 break;
166             } else if (c == quote) {
167                 if (src.peekNext() == quote) {
168                     src.eat();
169                     builder.append(quote);
170                 } else {
171                     break;
172                 }
173             } else if (c == separator) {
174                 if (quoted) {
175                     builder.append(c);
176                 } else {
177                     break;
178                 }
179             } else if (c == valueSeparator) {
180                 break;
181             } else if (c == escape) {
182                 src.eat();
183                 c = src.peek();
184                 if (c == Source.END) {
185                     throw new CsvLineException("tailing escape character " + escape);
186                 }
187                 builder.append((char) c);
188             } else {
189                 builder.append((char) c);
190             }
191             src.eat();
192         }
193         values.add(builder.toString());
194     }

198     public String write(Line line) {
199         StringWriter dest;
200         
201         dest = new StringWriter();
202         try {
203             write(line, dest);
204         } catch (IOException e) {
205             throw new RuntimeException("unused", e);
206         }
207         return dest.toString();
208     }
209     
210     public void write(Line line, Writer dest) throws IOException {
211         for (int i = 0, max = line.size(); i < max; i++) {
212             if (i > 0) {
213                 dest.write(separator);
214             }
215             writeCell(line.get(i), dest);
216         }
217         dest.write('\n');
218     }
219     
220     private void writeCell(List<String> values, Writer dest) throws IOException {
221         boolean quoted;
222         char c;
223         boolean first;
224 
225         if (values == null) {
226             dest.write(nul);
227             return;
228         }
229         if (values.size() == 0) {
230             dest.write(empty);
231             return;
232         }
233         quoted = values.size() != 1 || needsQuotes(values.get(0));
234         if (quoted) {
235             dest.write(quote);
236         }
237         first = true;
238         for (String value : values) {
239             if (first) {
240                 first = false;
241             } else {
242                 dest.write(valueSeparator);
243             }
244             if (value.equals(nul)) {
245                 dest.write(escape);
246                 dest.write(nul);
247             } else if (value.equals(empty)) {
248                 dest.write(escape);
249                 dest.write(empty);
250             } else {
251                 for (int i = 0, max = value.length(); i < max; i++) {
252                     c = value.charAt(i);
253                     if (c == quote) { 
254                         dest.write(quote);
255                         dest.write(quote);
256                     } else if (c == separator || c == valueSeparator || c == escape) {
257                         dest.write(escape);
258                         dest.write(c);
259                     } else {
260                         dest.write(c);
261                     }
262                 }
263             }
264         }
265         if (quoted) {
266             dest.write(quote);
267         }
268     }
269 
270     private boolean needsQuotes(String value) {
271         for (int i = 0, max = value.length(); i < max; i++) {
272             if (unquoted.indexOf(value.charAt(i)) == -1) {
273                 return true;
274             }
275         }
276         return false;
277     }
