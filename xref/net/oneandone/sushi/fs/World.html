49  /**
50   * <p>Configures and creates nodes. You'll usually create a single world instance in your application, configure it and
51   * afterwards use it through-out your application to create nodes via World.node or World.file. </p>
52   *
53   * <p>Sushi's FS subsystem forms a tree: An world object is the root, having filesystems as it's children, roots as
54   * grand-children and nodes as leaves. This tree is traversable from nodes up to the world object via Node.getRoot(),
55   * Root.getFilesystem() and Filesystem.getWorld(), which is used internally e.g. to pick default encoding settings
56   * from world. (Traversing in reverse order is not implemented, that too resource consuming)</p>
57   *
58   * <p>Technically, you can creates as many world objects as you which, but nodes from different world instances
59   * cannot interact, so you'll usually stick with a single world instance. Note that there's no
60   * predefined static world instance in Sushi. You can either get the world from node instances, or,
61   * if you feel a single world instance should be available everywhere in your application use dependency injection. </p>
62   *
63   * <p>TODO: Multi-threading. Currently, you need to know fs system internals to properly synchronized ...</p>
64   */

167         return this;

170     public <T extends Filesystem> T addFilesystem(T filesystem) {
171     	String name;
172 
173     	name = filesystem.getScheme();
174         if (filesystems.containsKey(name)) {
175             throw new IllegalArgumentException("duplicate filesystem scheme: " + name);
176         }
177         filesystems.put(name, filesystem);
178         return filesystem;
179     }

181     public Filesystem addFilesystemOpt(String filesystemClass, Object ... args) {
182         Class<?> clazz;
183         Constructor<?> constructor;
184         Filesystem filesystem;
185 
186         try {
187             clazz = Class.forName(filesystemClass);
188         } catch (NoClassDefFoundError e) {
189             return null;
190         } catch (ClassNotFoundException e) {
191             throw new IllegalArgumentException("class not found: " + filesystemClass, e);
192         }
193         try {
194             constructor = null;
195             for (Constructor<?> c : clazz.getConstructors()) {
196                 if (Reflect.matches(c.getParameterTypes(), args)) {
197                     if (constructor != null) {
198                         throw new IllegalArgumentException("constructor ambiguous");
199                     }
200                     constructor = c;
201                 }
202             }
203             if (constructor == null) {
204                 throw new IllegalArgumentException("no constructor: " + filesystemClass);
205             }
206             try {
207                 filesystem = (Filesystem) constructor.newInstance(args);
208             } catch (InvocationTargetException e) {
209                 throw new IllegalStateException("cannot create file system", e.getTargetException());
210             }
211             addFilesystem(filesystem);
212             return filesystem;
213         } catch (RuntimeException e) {
214             throw e;
215         } catch (Exception e) {
216             throw new IllegalArgumentException("cannot instantiate " + filesystemClass, e);
217         }
218     }

238     public <T extends Filesystem> T getFilesystem(String scheme, Class<T> clazz) {
239         Filesystem filesystem;
240 
241         filesystem = lookupFilesystem(scheme);
242         if (filesystem == null) {
243             throw new IllegalArgumentException("no such filesystem: " + scheme);
244         }
245         if (!clazz.isInstance(filesystem)) {
246             throw new IllegalArgumentException("unexpected file system type: " + filesystem.getClass().getName());
247         }
248         return (T) filesystem;
249     }

261     public FileNode file(String filePath) {
262         FileRoot root;
263         String path;
264 
265         root = fileFilesystem.lookupRoot(filePath);
266         if (root != null) {
267             filePath = filePath.substring(root.getAbsolute().length());
268         }
269         filePath = Strings.removeRightOpt(filePath, File.separator);
270    		path = filePath.replace(File.separatorChar, Filesystem.SEPARATOR_CHAR);
271         if (root == null) {
272             if (working == null) {
273                 throw new IllegalStateException("working directory is missing");
274             } else if (working instanceof FileNode) {
275                 return ((FileNode) working).join(path);
276             } else {
277                 throw new IllegalStateException("working directory is not a file: " + working.getURI());
278             }
279         } else {
280             return root.node(path, null);
281         }
282     }

303     public Node node(URI uri, Object extra) throws NodeInstantiationException {
304         String scheme;
305         Filesystem fs;
306         Node result;
307 
308         if (!uri.isAbsolute()) {
309             // because I cannot reliably create relative URIs. E.g. paths with ':' will always result a URI with scheme
310             throw new IllegalStateException("cannot create node for relative URI, use getWorking().join instead: " + uri);
311         }
312         scheme = uri.getScheme();
313         if (scheme == null) {
314             throw new IllegalStateException("missing scheme: " + uri);
315         }
316         fs = filesystems.get(scheme);
317         if (fs == null) {
318             throw new NodeInstantiationException(uri, "unknown scheme: " + scheme);
319         }
320         result = fs.node(uri, extra);
321         return result;
322     }

344     /** @param name must not start with a slash */
345     public Node resource(String name) throws IOException {
346         List<Node> result;
347 
348         result = resources(name);
349         switch (result.size()) {
350         case 0:
351             throw new FileNotFoundException("no such resource: " + name);
352         case 1:
353             return result.get(0);
354         default:
355             throw new IOException("resource ambiguous: " + name + "(" + result + ")");
356         }
357     }

372     /** @param name must not start with a slash */
373     private List<Node> resources(String name, boolean rejectDuplicates) throws IOException {
374         Enumeration<URL> e;
375         List<Node> result;
376         Node add;
377 
378         if (name.startsWith("/")) {
379             throw new IllegalArgumentException();
380         }
381         e = getClass().getClassLoader().getResources(name);
382         result = new ArrayList<>();
383         while (e.hasMoreElements()) {
384             try {
385                 add = node(e.nextElement().toURI());
386             } catch (URISyntaxException ex) {
387                 throw new IllegalStateException(ex);
388             }
389             if (result.contains(add)) {
390                 if (rejectDuplicates) {
391                     throw new IllegalStateException("duplicate classpath item: " + add);
392                 }
393             } else {
394                 result.add(add);
395             }
396         }
397         return result;
398     }

466     /** Throws a RuntimeException if the resource is not found */
467     public FileNode locateClasspathItem(Class<?> base, String resourcename) {
468         URL url;
469         FileNode file;
470 
471         url = base.getResource(resourcename);
472         if (url == null) {
473             throw new RuntimeException("no such resource: " + resourcename);
474         }
475         file = locateClasspathItem(url, resourcename);
476         if (!file.exists()) {
477             throw new RuntimeException(url + ": no such file or directory: " + file);
478         }
479         file = locateClasspathItem(url, resourcename);
480         if (!file.exists()) {
481             throw new RuntimeException(url + ": no such file or directory: " + file);
482         }
483         return file;
484     }

486     public FileNode guessProjectHome(Class<?> c) {
487         FileNode node;
488 
489         node = locateClasspathItem(c);
490         if (node.isDirectory()) {
491             if (node.getName().endsWith("classes")) {
492                 node = node.getParent();
493             }
494         } else {
495             if (node.getName().endsWith(".jar")) {
496                 node = node.getParent();
497             }
498         }
499         if (node.getName().endsWith("target")) {
500             node = node.getParent();
501         }
502         return node;
503     }

505     /**
506      * Returns the file of a certain class at a special location. e.g. jar files
507      *
508      * @param url the destination path to the resource
509      * @param resourcename  absolute resource name; redundant, but necessary to strip from urls
510      *
511      * @return the physical file referring to the class
512      */
513     public FileNode locateClasspathItem(URL url, String resourcename) {
514         String filename;
515         FileNode file;
516         String protocol;
517         int idx;
518 
519         if (!resourcename.startsWith("/")) {
520             throw new IllegalArgumentException("absolute resourcename expected: " + resourcename);
521         }
522         protocol = url.getProtocol();
523         if ("file".equals(protocol)) {
524             try {
525                 file = file(new File(url.toURI()));
526             } catch (URISyntaxException e) {
527                 throw new IllegalStateException(e);
528             }
529             filename = file.getAbsolute();
530             if (!filename.endsWith(resourcename.replace('/', File.separatorChar))) {
531                 throw new RuntimeException("classname not found in file url: " + filename + " " + resourcename);
532             }
533             file = file(filename.substring(0, filename.length() - resourcename.length()));
534         } else if ("jar".equals(protocol)) {
535             filename = url.getFile();
536             idx = filename.indexOf("!/");
537             if (idx == -1) {
538                 throw new RuntimeException("!/ not found: " + filename);
539             }
540             filename = filename.substring(0, idx);
541             try {
542 				file = (FileNode) node(filename);
543 			} catch (NodeInstantiationException | URISyntaxException e) {
544 				throw new IllegalStateException(filename, e);
545 			}
546         } else {
547             throw new RuntimeException("protocol not supported: " + protocol);
548         }
549         return file;
550     }

554     private FileNode init(String name) {
555         String value;
556         File file;
557 
558         value = System.getProperty(name);
559         if (value == null) {
560             throw new IllegalStateException("property not found: " + name);
561         }
562         file = new File(value);
563         if (!file.isDirectory()) {
564             throw new IllegalStateException(
565                 "property " + name + " does not point to a directory: " + value);
566         }
567         return file(file);
568     }
