29  /**
30   * <p>Similar to java.world.FileFilter or Ant File/Directory sets. A filter is basically a list of paths to
31   * include or exclude. Predicates can be used to further restrict the collected nodes. Paths always use
32   * slashes (/) - even on Windows - because a) it simplifies writings constants and b) a filter can be
33   * applied to any file system.
34   *
35   * <p>Usage. Create a new instance, use the various selections methods (include, exclude, etc.),
36   * and pass the instance for dir.find(). Selection methods return <code>this</code> to allow expressions.</p>
37   *
38   * <p>A path is a list of names, separated by Filter.SEPARATOR. Paths must *not* start with a
39   * separator, i.e they have to be relative. Paths remain relative until the filter is actually
40   * applied to a tree. Paths must not end with a separator either. </p>
41   *
42   * <p>Names use the familiar glob syntax. Filters do not know about extensions. </p>
43   */

174     private Object[] compile(String path) {
175         List<String> lst;
176 
177         lst = Filesystem.SEPARATOR.split(path);
178         if (lst.size() == 0) {
179             throw new IllegalArgumentException("empty path: " + path);
180         }
181         if (lst.get(0).equals("")) {
182             throw new IllegalArgumentException("absolute path not allowed: " + path);
183         }
184         if (lst.get(lst.size() - 1).equals("")) {
185             throw new IllegalArgumentException(
186                 "path must not end with separator: " + path);
187         }
188         return compileTail(lst, 0);
189     }
190 
191     /**
192      * @param lst  array of patterns
193      */
194     private Object[] compileTail(List<String> lst, int start) {
195         Object head;
196         Object[] tail;
197 
198         if (start == lst.size()) {
199             return null;
200         } else {
201             head = Glob.compile(lst.get(start), ignoreCase);
202             tail = compileTail(lst, start + 1);
203             if (head == Glob.STARSTAR) {
204                 if (tail == null) {
205                     throw new IllegalArgumentException("** must be followed by some content");
206                 }
207                 if (tail[0] == Glob.STARSTAR) {
208                     throw new IllegalArgumentException("**/** is not allowed");
209                 }
210             }
211         }
212         return new Object[] { head, tail };
213     }

236     private void doInvoke(int currentDepth, Node parent, boolean parentIsLink, List<Object[]> includes, List<Object[]> excludes, Action result)
237     throws IOException {
238         List<? extends Node> children;
239         List<Object[]> remainingIncludes;
240         List<Object[]> remainingExcludes;
241         String name;
242         boolean childIsLink;
243         boolean in;
244         boolean ex;
245 
246         if (currentDepth >= maxDepth) {
247             return;
248         }
249         if (!followLinks && parentIsLink) {
250             return;
251         }
252         try {
253             children = list(parent, includes);
254         } catch (IOException e) {
255             result.enterFailed(parent, parentIsLink, e);
256             return;
257         }
258         if (children == null) {
259             // ignore file
260         } else {
261             result.enter(parent, parentIsLink);
262             currentDepth++;
263             for (Node child : children) {
264                 name = child.getName();
265                 childIsLink = child.isLink();
266                 remainingIncludes = new ArrayList<Object[]>();
267                 remainingExcludes = new ArrayList<Object[]>();
268                 in = doMatch(name, includes, remainingIncludes);
269                 ex = doMatch(name, excludes, remainingExcludes);
270                 if (in && !ex && currentDepth >= minDepth && matchPredicates(child, childIsLink)) {
271                     result.select(child, childIsLink);
272                 }
273                 if (remainingIncludes.size() > 0 && !excludesAll(remainingExcludes)) {
274                     doInvoke(currentDepth, child, childIsLink, remainingIncludes, remainingExcludes, result);
275                 }
276             }
277             result.leave(parent, parentIsLink);
278         }
279     }

281     // avoids node.list() call with there is exactly 1 include with a literal head
282     private List<? extends Node> list(Node node, List<Object[]> includes) throws IOException {
283     	Node child;
284 
285     	if (includes.size() == 1 && includes.get(0)[0] instanceof String) {
286             child = node.join((String) includes.get(0)[0]);
287             if (child.exists()) {
288                 return Collections.singletonList(child);
289             } else {
290                 return Collections.emptyList();
291             }
292     	} else {
293         	return node.list();
294         }
295     }

297     private boolean matchPredicates(Node node, boolean isLink) throws IOException {
298         for (Predicate p : predicates) {
299             if (!p.matches(node, isLink)) {
300                 return false;
301             }
302         }
303         return true;
304     }

306     private static boolean excludesAll(List<Object[]> excludes) {
307         int i;
308         int max;
309         Object[] pair;
310         Object[] tail;
311 
312         max = excludes.size();
313         for (i = 0; i < max; i++) {
314             pair = excludes.get(i);
315             tail = (Object[]) pair[1];
316             if (pair[0] == Glob.STARSTAR && tail[0] == Glob.STAR) {
317                 return true;
318             }
319         }
320         return false;
321     }

323     private static boolean doMatch(String name, List<Object[]> paths, List<Object[]> remainingPaths) {
324         boolean found;
325         int i;
326         int max;
327         Object[] path;
328         Object head;
329         Object[] tail;
330 
331         found = false;
332         max = paths.size();
333         for (i = 0; i < max; i++) {
334             path = paths.get(i);
335             if (path == null) {
336                 throw new IllegalStateException("unexpected empty path");
337             }
338             head = path[0];
339             tail = (Object[]) path[1];
340             if (head == Glob.STARSTAR) {
341                 remainingPaths.add(path);
342                 head = tail[0];
343                 tail = (Object[]) tail[1];
344             }
345             if (matches(head, name)) {
346                 if (tail != null) {
347                     remainingPaths.add(tail);
348                 } else {
349                     found = true;
350                 }
351             }
352         }
353         return found;
354     }
355     private static boolean matches(Object stringOrPattern, String name) {
356     	if (stringOrPattern instanceof String) {
357     		return name.equals(stringOrPattern);
358     	} else {
359     		return Glob.matches((Pattern) stringOrPattern, name);
360     	}
361     }
