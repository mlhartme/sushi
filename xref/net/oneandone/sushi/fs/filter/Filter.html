29  /**
30   * <p>Similar to java.world.FileFilter or Ant File/Directory sets. A filter is basically a list of paths to
31   * include or exclude. Predicates can be used to further restrict the collected nodes. Paths always use
32   * slashes (/) - even on Windows - because a) it simplifies writings constants and b) a filter can be
33   * applied to any file system.
34   *
35   * <p>Usage. Create a new instance, use the various selections methods (include, exclude, etc.),
36   * and pass the instance for dir.find(). Selection methods return <code>this</code> to allow expressions.</p>
37   *
38   * <p>A path is a list of names, separated by Filter.SEPARATOR. Paths must *not* start with a
39   * separator, i.e they have to be relative. Paths remain relative until the filter is actually
40   * applied to a tree. Paths must not end with a separator either. </p>
41   *
42   * <p>Names use the familiar glob syntax. Filters do not know about extensions. </p>
43   */
44  public class Filter {
45      public static final Filter NOTHING = new Filter().exclude("**/*");
46      public static final int DEPTH_INFINITE = Integer.MAX_VALUE;

127     public Filter include(List<String> paths) {
128         for (String path : paths) {

175     private Object[] compile(String path) {
176         List<String> lst;
177 
178         lst = Filesystem.SEPARATOR.split(path);
179         if (lst.size() == 0) {
180             throw new IllegalArgumentException("empty path: " + path);
181         }
182         if (lst.get(0).equals("")) {
183             throw new IllegalArgumentException("absolute path not allowed: " + path);
184         }
185         if (lst.get(lst.size() - 1).equals("")) {
186             throw new IllegalArgumentException(
187                 "path must not end with separator: " + path);
188         }
189         return compileTail(lst, 0);
190     }
191 
192     /**
193      * @param lst  array of patterns
194      */
195     private Object[] compileTail(List<String> lst, int start) {
196         Object head;
197         Object[] tail;
198 
199         if (start == lst.size()) {
200             return null;
201         } else {
202             head = Glob.compile(lst.get(start), ignoreCase);
203             tail = compileTail(lst, start + 1);
204             if (head == Glob.STARSTAR) {
205                 if (tail == null) {
206                     throw new IllegalArgumentException("** must be followed by some content");
207                 }
208                 if (tail[0] == Glob.STARSTAR) {
209                     throw new IllegalArgumentException("**/** is not allowed");
210                 }
211             }
212         }
213         return new Object[] { head, tail };
214     }

228     /**
229      * Tests includes an excludes. CAUTION: does not support checks that need a node (like predicates). Ignores "followSymlinks"
230      */
231     public boolean matches(String path) {
232         List<String> segments;
233 
234         segments = Filesystem.SEPARATOR.split(path);
235         if (segments.size() < minDepth || segments.size() > maxDepth) {
236             return false;
237         }
238         if (predicates.size() > 0) {
239             throw new UnsupportedOperationException("cannot match with predicates");
240         }
241         return matches(0, segments, new ArrayList<>(includes), new ArrayList<>(excludes));
242     }
243 
244     private boolean matches(int currentSegment, List<String> segments, List<Object[]> includes, List<Object[]> excludes) {
245         List<Object[]> remainingIncludes;
246         List<Object[]> remainingExcludes;
247         String name;
248         boolean in;
249         boolean ex;
250 
251         if (currentSegment >= segments.size()) {
252             return false;
253         }
254         name = segments.get(currentSegment);
255         remainingIncludes = new ArrayList<>();
256         remainingExcludes = new ArrayList<>();
257         in = doMatch(name, includes, remainingIncludes);
258         ex = doMatch(name, excludes, remainingExcludes);
259         if (in && !ex) {
260             return true;
261         }
262         if (remainingIncludes.size() > 0 && !excludesAll(remainingExcludes)) {
263             return matches(currentSegment + 1, segments, remainingIncludes, remainingExcludes);
264         } else {
265             return false;
266         }
267     }

269     /**
270      * Main methods of this class.
271      *
272      * @throws IOException as thrown by the specified FileTask
273      */
274     public void invoke(Node root, Action result) throws IOException {
275         doInvoke(0, root, root.isLink(), new ArrayList<>(includes), new ArrayList<>(excludes), result);
276     }

278     private void doInvoke(int currentDepth, Node parent, boolean parentIsLink, List<Object[]> includes, List<Object[]> excludes, Action result)
279     throws IOException {
280         List<? extends Node> children;
281         List<Object[]> remainingIncludes;
282         List<Object[]> remainingExcludes;
283         String name;
284         boolean childIsLink;
285         boolean in;
286         boolean ex;
287 
288         if (currentDepth >= maxDepth) {
289             return;
290         }
291         if (!followLinks && parentIsLink) {
292             return;
293         }
294         try {
295             children = list(parent, includes);
296         } catch (IOException e) {
297             result.enterFailed(parent, parentIsLink, e);
298             return;
299         }
300         if (children == null) {
301             // ignore file
302         } else {
303             result.enter(parent, parentIsLink);
304             currentDepth++;
305             for (Node child : children) {
306                 name = child.getName();
307                 childIsLink = child.isLink();
308                 remainingIncludes = new ArrayList<>();
309                 remainingExcludes = new ArrayList<>();
310                 in = doMatch(name, includes, remainingIncludes);
311                 ex = doMatch(name, excludes, remainingExcludes);
312                 if (in && !ex && currentDepth >= minDepth && matchPredicates(child, childIsLink)) {
313                     result.select(child, childIsLink);
314                 }
315                 if (remainingIncludes.size() > 0 && !excludesAll(remainingExcludes)) {
316                     doInvoke(currentDepth, child, childIsLink, remainingIncludes, remainingExcludes, result);
317                 }
318             }
319             result.leave(parent, parentIsLink);
320         }
321     }

323     // avoids node.list() call if there is exactly 1 include with a literal head
324     private List<? extends Node> list(Node node, List<Object[]> includes) throws IOException {
325     	Node child;
326 
327     	if (includes.size() == 1 && includes.get(0)[0] instanceof String) {
328             child = node.join((String) includes.get(0)[0]);
329             if (child.exists()) {
330                 return Collections.singletonList(child);
331             } else {
332                 return Collections.emptyList();
333             }
334     	} else {
335         	return node.list();
336         }
337     }

339     private boolean matchPredicates(Node node, boolean isLink) throws IOException {
340         for (Predicate p : predicates) {
341             if (!p.matches(node, isLink)) {
342                 return false;
343             }
344         }
345         return true;
346     }

348     private static boolean excludesAll(List<Object[]> excludes) {
349         int i;
350         int max;
351         Object[] pair;
352         Object[] tail;
353 
354         max = excludes.size();
355         for (i = 0; i < max; i++) {
356             pair = excludes.get(i);
357             tail = (Object[]) pair[1];
358             if (pair[0] == Glob.STARSTAR && tail[0] == Glob.STAR) {
359                 return true;
360             }
361         }
362         return false;
363     }
364 
365     private static boolean doMatch(String name, List<Object[]> paths, List<Object[]> remainingPaths) {
366         boolean found;
367         int i;
368         int max;
369         Object[] path;
370         Object head;
371         Object[] tail;
372 
373         found = false;
374         max = paths.size();
375         for (i = 0; i < max; i++) {
376             path = paths.get(i);
377             if (path == null) {
378                 throw new IllegalStateException("unexpected empty path");
379             }
380             head = path[0];
381             tail = (Object[]) path[1];
382             if (head == Glob.STARSTAR) {
383                 remainingPaths.add(path);
384                 head = tail[0];
385                 tail = (Object[]) tail[1];
386             }
387             if (matches(head, name)) {
388                 if (tail != null) {
389                     remainingPaths.add(tail);
390                 } else {
391                     found = true;
392                 }
393             }
394         }
395         return found;
396     }
397     private static boolean matches(Object stringOrPattern, String name) {
398     	if (stringOrPattern instanceof String) {
399     		return name.equals(stringOrPattern);
400     	} else {
401     		return Glob.matches((Pattern) stringOrPattern, name);
402     	}
403     }
