16  package net.oneandone.sushi.fs.memory;
17  
18  import net.oneandone.sushi.fs.Filesystem;
19  import net.oneandone.sushi.fs.LengthException;
20  import net.oneandone.sushi.fs.Root;
21  import net.oneandone.sushi.fs.file.FileNode;
22  import net.oneandone.sushi.io.CheckedByteArrayInputStream;
23  
24  import java.io.IOException;
25  import java.io.InputStream;
26  import java.util.ArrayList;
27  import java.util.Arrays;
28  import java.util.HashMap;
29  import java.util.List;
30  import java.util.Map;
31  
32  public class MemoryRoot implements Root<MemoryNode> {
33      public final MemoryFilesystem filesystem;
34      public final int id;
35      private final Map<String, MemoryNode> nodes;
36      private final Map<String, Object> store;
37      
38      public MemoryRoot(MemoryFilesystem filesystem, int id) {
39          this.filesystem = filesystem;
40          this.id = id;
41          this.nodes = new HashMap<String, MemoryNode>();
42          this.store = new HashMap<String, Object>();
43          add(new MemoryNode(this, "", Type.DIRECTORY));
44      }
45  
46      public MemoryFilesystem getFilesystem() {
47          return filesystem;
48      }
49  
50      public String getId() {
51          return "//" + id + "/";
52      }
53  
54      public MemoryNode node(String path, String encodedQuery) {
55          MemoryNode node;
56          
57          if (encodedQuery != null) {
58              throw new IllegalArgumentException(encodedQuery);
59          }
60          node = nodes.get(path);
61          if (node == null) {
62              node = new MemoryNode(this, path, Type.NONE);
63              nodes.put(node.getPath(), node);
64          }
65          return node;
66      }
67      
68      public void add(MemoryNode node) {
69          nodes.put(node.getPath(), node);
70      }
71      
72      public long length(String path) throws LengthException {
73          Object obj;
74  
75          obj = store.get(path);
76          if (obj instanceof FileNode) {
77              return ((FileNode) obj).length();
78          } else {
79              return ((byte[]) obj).length;
80          }
81      }
82      
83      public List<MemoryNode> list(String path) {
84          String child;
85          int idx;
86          List<MemoryNode> result;
87          
88          result = new ArrayList<MemoryNode>();
89          for (MemoryNode node : nodes.values()) {
90              child = node.getPath();
91              idx = child.lastIndexOf(Filesystem.SEPARATOR_CHAR);
92              if (!path.equals(child) && path.equals(idx == -1 ? "" : child.substring(0, idx))) {
93                  if (node.exists()) {
94                      result.add(node);
95                  }
96              }
97          }     
98          return result;
99      }
100     
101     //--
102     
103     InputStream open(String path) throws IOException {
104         Object obj;
105         
106         obj = store.get(path);
107         if (obj instanceof FileNode) {
108             return ((FileNode) obj).createInputStream();
109         } else {
110             return new CheckedByteArrayInputStream((byte[]) obj);
111         }
112     }
113 
114     byte[] readBytes(String path) throws IOException {
115         Object obj;
116         byte[] bytes;
117 
118         obj = store.get(path);
119         if (obj instanceof FileNode) {
120             return ((FileNode) obj).readBytes();
121         } else {
122             bytes = (byte[]) obj;
123             return Arrays.copyOf(bytes, bytes.length);
124         }
125     }
126     
127     void store(String path, byte[] data, int used) throws IOException {
128         Object old;
129         FileNode file;
130         byte[] copy;
131         
132         old = store.get(path);
133         if (old instanceof FileNode) {
134             ((FileNode) old).deleteTree();
135         }
136         if (used > filesystem.getMaxInMemorySize()) {
137             file = filesystem.getWorld().getTemp().createTempFile();
138             file.writeBytes(data, 0, used, false);
139             store.put(path, file);
140         } else {
141             copy = new byte[used];
142             System.arraycopy(data, 0, copy, 0, used);
143             store.put(path, copy);
144         }
145     }
146 }
