34  /** Nodes accessible via svn. Uses svnkit for server-side communication, but not to interact with local checkouts. */
35  public class SvnFilesystem extends Filesystem {
36      static {
37          FSRepositoryFactory.setup();
38          DAVRepositoryFactory.setup();
39          System.setProperty("svnkit.upgradeWC", "false"); // see https://wiki.svnkit.com/SVNKit_specific_system_properties
40      }

42      private ISVNAuthenticationManager defaultAuthenticationManager;
43  
44      public SvnFilesystem(World world, String name) {
45          super(world, new Features(true, false, false, false, false, false, true), name);
46  
47          this.defaultAuthenticationManager = authenticationManager(null, null);
48      }
49  
50      public ISVNAuthenticationManager getDefaultAuthenticationManager() {
51          return defaultAuthenticationManager;
52      }
53  
54      public void setDefaultCredentials(String username, String password) {
55          setDefaultAuthenticationManager(authenticationManager(username, password));
56      }
57  
58      public void setDefaultAuthenticationManager(ISVNAuthenticationManager authenticationManager) {
59          this.defaultAuthenticationManager = authenticationManager;
60      }

63      @Override
64      public SvnNode node(URI uri, Object extra) throws NodeInstantiationException {
65          String encodedSchemeSpecific;
66          String encodedPath;
67          String separator;
68          String root;
69          SVNRepository repository;
70  
71          if (extra != null) {
72              throw new NodeInstantiationException(uri, "unexpected extra argument: " + extra);
73          }
74          checkOpaque(uri);
75          separator = SEPARATOR_STRING;
76          encodedSchemeSpecific = uri.getRawSchemeSpecificPart();
77          try {
78              repository = repository(encodedSchemeSpecific);
79              root = repository.getRepositoryRoot(true).toString();
80              if (!encodedSchemeSpecific.startsWith(root)) {
81                  throw new IllegalStateException(encodedSchemeSpecific + " vs " + root);
82              }
83              encodedPath = encodedSchemeSpecific.substring(root.length());
84              if (encodedPath.length() > 0) {
85                  if (!encodedPath.startsWith(separator)) {
86                      throw new IllegalStateException(encodedSchemeSpecific + " vs " + root);
87                  }
88                  encodedPath = encodedPath.substring(separator.length());
89              }
90              if (encodedPath.endsWith(separator)) {
91                  throw new NodeInstantiationException(uri, "invalid tailing " + separator);
92              }
93              if (encodedPath.startsWith(separator)) {
94                  throw new NodeInstantiationException(uri, "invalid heading " + separator);
95              }
96              repository = repository(encodedSchemeSpecific.substring(0, encodedSchemeSpecific.length() - encodedPath.length()));
97              return new SvnRoot(this, repository).node(SVNEncodingUtil.uriDecode(encodedPath), null);
98          } catch (SVNException e) {
99              throw new NodeInstantiationException(uri, e.getMessage(), e);
100         }
101     }

105     public SVNRepository repository(String urlstr) throws SVNException {
106         ISVNAuthenticationManager authenticationManager;
107         String userinfo;
108         SVNURL url;
109         int idx;
110         String username;
111         String password;
112 
113         url = SVNURL.parseURIEncoded(urlstr);
114         userinfo = url.getUserInfo();
115         if (userinfo != null) {
116             idx = userinfo.indexOf(':');
117             if (idx == -1) {
118                 username = userinfo;
119                 password = null;
120             } else {
121                 username = userinfo.substring(0, idx);
122                 password = userinfo.substring(idx + 1);
123             }
124             authenticationManager = authenticationManager(username, password);
125         } else {
126             authenticationManager = defaultAuthenticationManager == null ? authenticationManager(null, null) : defaultAuthenticationManager;
127         }
128 
129         return repository(url, authenticationManager);
130     }

134     public static SVNRepository repository(SVNURL url, ISVNAuthenticationManager authenticationManager) throws SVNException {
135         SVNRepository repository;
136 
137         repository = SVNRepositoryFactory.create(url);
138         repository.setAuthenticationManager(authenticationManager);
139         return repository;
140     }
141 
142     public static ISVNAuthenticationManager authenticationManager(String username, String password) {
143         return SVNWCUtil.createDefaultAuthenticationManager(
144                 SVNWCUtil.getDefaultConfigurationDirectory(),
145                 username, password,
146                 false /* do not store credentials, not even when configured */);
147     }
