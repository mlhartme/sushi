19  import org.apache.http.HttpConnectionMetrics;
20  import org.apache.http.HttpEntityEnclosingRequest;
21  import org.apache.http.HttpException;
22  import org.apache.http.HttpRequest;
23  import org.apache.http.HttpResponse;
24  import org.apache.http.impl.DefaultHttpResponseFactory;
25  import org.apache.http.impl.HttpConnectionMetricsImpl;
26  import org.apache.http.impl.entity.EntityDeserializer;
27  import org.apache.http.impl.entity.EntitySerializer;
28  import org.apache.http.impl.entity.LaxContentLengthStrategy;
29  import org.apache.http.impl.entity.StrictContentLengthStrategy;
30  import org.apache.http.impl.io.HttpRequestWriter;
31  import org.apache.http.impl.io.HttpResponseParser;
32  import org.apache.http.impl.io.SocketInputBuffer;
33  import org.apache.http.impl.io.SocketOutputBuffer;
34  import org.apache.http.io.HttpMessageParser;
35  import org.apache.http.io.HttpMessageWriter;
36  import org.apache.http.params.HttpConnectionParams;
37  import org.apache.http.params.HttpParams;
38  
39  import java.io.IOException;
40  import java.net.Socket;
41  import java.net.SocketException;
42  import java.util.logging.Level;
43  
44  public class WebdavConnection implements HttpClientConnection {
45      public static WebdavConnection open(Socket socket, HttpParams params) throws IOException {
46          int linger;
47          int buffersize;
48          SocketInputBuffer input;
49          SocketOutputBuffer output;
50          
51          socket.setTcpNoDelay(HttpConnectionParams.getTcpNoDelay(params));
52          socket.setSoTimeout(HttpConnectionParams.getSoTimeout(params));
53          linger = HttpConnectionParams.getLinger(params);
54          if (linger >= 0) {
55              socket.setSoLinger(linger > 0, linger);
56          }
57          buffersize = HttpConnectionParams.getSocketBufferSize(params);
58  		if (WebdavFilesystem.WIRE.isLoggable(Level.FINE)) {
59  			input = new LoggingSessionInputBuffer(socket, buffersize, params, WebdavFilesystem.WIRE);
60              output = new LoggingSessionOutputBuffer(socket, buffersize, params, WebdavFilesystem.WIRE);
61  		} else {
62  	        input = new SocketInputBuffer(socket, buffersize, params);
63              output = new SocketOutputBuffer(socket, buffersize, params);
64  		}
65          return new WebdavConnection(socket, input, output, params);
66      }
67  
68      private final Socket socket;
69      private final EntitySerializer entityserializer;
70      private final EntityDeserializer entitydeserializer;
71      private final SocketInputBuffer input;
72      private final SocketOutputBuffer output;
73      private final HttpMessageParser responseParser;
74      private final HttpMessageWriter requestWriter;
75      private final HttpConnectionMetricsImpl metrics;
76  	private boolean open;
77  
78      public WebdavConnection(Socket socket, SocketInputBuffer input, SocketOutputBuffer output, HttpParams params) {
79      	this.socket = socket;
80          this.entityserializer = new EntitySerializer(new StrictContentLengthStrategy());
81          this.entitydeserializer = new EntityDeserializer(new LaxContentLengthStrategy());
82          this.input = input;
83          this.output = output;
84          this.responseParser = new HttpResponseParser(input, null, new DefaultHttpResponseFactory(), params);
85          this.requestWriter = new HttpRequestWriter(output, null, params);
86          this.metrics = new HttpConnectionMetricsImpl(input.getMetrics(), output.getMetrics());
87          this.open = true;
88      }
89  
90      //-- HttpClientConnection implementation
91      
92      public boolean isResponseAvailable(int timeout) throws IOException {
93          return input.isDataAvailable(timeout);
94      }
95  
96      public void sendRequestHeader(final HttpRequest request) throws HttpException, IOException {
97          requestWriter.write(request);
98          metrics.incrementRequestCount();
99      }
100 
101     public void sendRequestEntity(final HttpEntityEnclosingRequest request) throws HttpException, IOException {
102         if (request.getEntity() != null) {
103         	entityserializer.serialize(output, request, request.getEntity());
104         }
105     }
106 
107     public HttpResponse receiveResponseHeader() throws HttpException, IOException {
108         HttpResponse response = (HttpResponse) responseParser.parse();
109         if (response.getStatusLine().getStatusCode() >= 200) {
110             metrics.incrementResponseCount();
111         }
112         return response;
113     }
114     
115     public void receiveResponseEntity(final HttpResponse response) throws HttpException, IOException {
116         response.setEntity(entitydeserializer.deserialize(input, response));
117     }
118     
119     public void flush() throws IOException {
120         output.flush();
121     }
122 
123     //-- HttpConnection implementation
124     
125     public void close() throws IOException {
126         if (!open) {
127             return;
128         }
129         open = false;
130         socket.close();
131     }
132     
133     public boolean isOpen() {
134         return open;
135     }
136     
137     public boolean isStale() {
138         if (!isOpen() || isEof()) {
139             return true;
140         }
141         try {
142             input.isDataAvailable(1);
143             return false;
144         } catch (IOException ex) {
145             return true;
146         }
147     }
148     
149     public void setSocketTimeout(int timeout) {
150         try {
151             socket.setSoTimeout(timeout);
152         } catch (SocketException e) {
153         	throw new RuntimeException("TODO", e);
154         }
155     }
156     
157     public int getSocketTimeout() {
158         try {
159             return socket.getSoTimeout();
160         } catch (SocketException e) {
161         	throw new RuntimeException("TODO", e);
162         }
163     }
164 
165     /** CAUTION: HttpCore and Java Sockets have a different terminology here */
166     public void shutdown() throws IOException {
167         open = false;
168         socket.close();
169     }
170     
171     public HttpConnectionMetrics getMetrics() {
172         return metrics;
173     }
174 
175     //--
176     
177     private boolean isEof() {
178         return input.isEof();
179     }
180     
181     public SocketOutputBuffer getOutputBuffer() {
182     	return output;
183     }
184     
185 	@Override
186     public String toString() {
187 		return "WebdavConnection(" + socket.getPort() + ')';
188     }
189 }
