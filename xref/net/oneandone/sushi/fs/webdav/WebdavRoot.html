41  public class WebdavRoot implements Root<WebdavNode> {
42      private final WebdavFilesystem filesystem;
43      public final HttpHost host;
44      private final HttpParams params;
45      private String authorization;
46  
47      public WebdavRoot(WebdavFilesystem filesystem, String protocol, String host, int port) {
48          this.filesystem = filesystem;
49          this.host = new HttpHost(host, port, protocol);
50          this.authorization = null;
51          this.params = new BasicHttpParams();
52  
53          HttpProtocolParams.setVersion(params, HttpVersion.HTTP_1_1);
54          HttpProtocolParams.setContentCharset(params, WebdavFilesystem.ENCODING);
55      }

57      public void setUserInfo(String userinfo) {
58          int idx;
59  
60          idx = userinfo.indexOf(':');
61          if (idx == -1) {
62              setCredentials(userinfo, "");
63          } else {
64              setCredentials(userinfo.substring(0, idx), userinfo.substring(idx + 1));
65          }
66      }
67  
68      public void setCredentials(String username, String password) {
69      	authorization = "Basic " + new Base64(true).run(username + ":" + password);
70      }

120     private final List<WebdavConnection> pool = new ArrayList<WebdavConnection>();
121     private int allocated = 0;
122 
123     public synchronized WebdavConnection allocate() throws IOException {
124         int size;
125 
126         allocated++;
127         size = pool.size();
128         if (size > 0) {
129             return pool.remove(size - 1);
130         } else {
131         	Socket socket;
132 
133             if ("https".equals(host.getSchemeName())) {
134                 socket = SSLSocketFactory.getDefault().createSocket(host.getHostName(), host.getPort());
135             } else {
136                 socket = new Socket(host.getHostName(), host.getPort());
137             }
138             return WebdavConnection.open(socket, params);
139         }
140     }
141 
142     public synchronized void free(HttpResponse response, WebdavConnection connection) throws IOException {
143         HttpEntity entity;
144 
145         if (allocated == 0) {
146             throw new IllegalStateException();
147         }
148         allocated--;
149         if (response != null) {
150             entity = response.getEntity();
151             if (entity != null) {
152                 entity.consumeContent();
153             }
154         }
155         if (wantsClose(response)) {
156             connection.close();
157         }
158         if (connection.isOpen() && pool.size() < 10) {
159             pool.add(connection);
160         }
161     }
162 
163     public synchronized int getAllocated() {
164         return allocated;
165     }
166 
167     private static boolean wantsClose(HttpResponse response) {
168         if (response == null) {
169             // no response yet
170             return true;
171         }
172         for (Header header : response.getHeaders(HTTP.CONN_DIRECTIVE)) {
173             if ("close".equalsIgnoreCase(header.getValue())) {
174                 return true;
175             }
176         }
177         return false;
178     }

182     public void send(WebdavConnection conn, HttpRequest request) throws IOException {
183         // TODO: side effect
184         request.addHeader(HTTP.TARGET_HOST, host.getHostName());
185         if (authorization != null) {
186             request.addHeader("Authorization", authorization);
187         }
188         // TODO: request.addHeader("Keep-Alive", "300");
189 
190         try {
191             conn.sendRequestHeader(request);
192             if (request instanceof HttpEntityEnclosingRequest) {
193                 conn.sendRequestEntity((HttpEntityEnclosingRequest) request);
194             }
195             conn.flush();
196         } catch (IOException e) {
197             free(null, conn);
198             throw e;
199         } catch (HttpException e) {
200             free(null, conn);
201             // TODO
202           	throw new IOException(e);
203         } catch (RuntimeException e) {
204             free(null, conn);
205             throw e;
206         }
207     }
208 
209     public HttpResponse receive(WebdavConnection conn, boolean head) throws IOException {
210         HttpResponse response;
211         int statuscode;
212 
213         response = null;
214         try {
215             do {
216                 response = conn.receiveResponseHeader();
217                 if (canResponseHaveBody(response, head)) {
218                     conn.receiveResponseEntity(response);
219                 }
220                 statuscode = response.getStatusLine().getStatusCode();
221             } while (response == null || statuscode < HttpStatus.SC_OK);
222             return response;
223         } catch (IOException e) {
224             free(response, conn);
225             throw e;
226         } catch (HttpException e) {
227             free(response, conn);
228             // TODO
229           	throw new IOException(e);
230         } catch (RuntimeException e) {
231             free(response, conn);
232             throw e;
233         }
234     }
235 
236     private boolean canResponseHaveBody(HttpResponse response, boolean head) {
237         int status;
238 
239         status = response.getStatusLine().getStatusCode();
240         if (status == HttpStatus.SC_OK && head) {
241             return false;
242         }
243         return status >= HttpStatus.SC_OK
244             && status != HttpStatus.SC_NO_CONTENT
245             && status != HttpStatus.SC_NOT_MODIFIED
246             && status != HttpStatus.SC_RESET_CONTENT;
247     }
