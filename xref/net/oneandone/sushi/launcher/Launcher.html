16  package net.oneandone.sushi.launcher;
17  
18  import net.oneandone.sushi.fs.file.FileNode;
19  import net.oneandone.sushi.util.Separator;
20  
21  import java.io.File;
22  import java.io.IOException;
23  import java.io.Reader;
24  import java.io.StringWriter;
25  import java.io.Writer;
26  import java.util.List;
27  
28  /**
29   * Configures and executes an operating system process. This class wraps a ProcessBuilder to simplify usage.
30   * In particular, most methods return this so you can configure and execute a program in a single expression
31   * (short methods names further simplify this). In addition, you can easily get process output as a string.
32   *
33   * None-zero exit codes of a process are reported as ExitCode exceptions. This helps to improve reliability
34   * because it's harder to ignore exceptions than to ignore return codes.
35   *
36   * Launcher streams chars, although the underlying process stream bytes. Encoding/decoding is performed
37   * according to the specified encoding. If you need to stream bytes with Launcher: pass in the appropriate
38   * OutputStreamWriter/InputStreamReader and make sure that their encoding and the encoding for Launcher is
39   * ascii. (Rationale: character stream are usually more convenient to use in applications than byte streams
40   * because process input/output is usually for humans - and thus has to be converted. To keep things simple,
41   * I accept the small performance penalty of also de- and encoding; if this is an issue in your application,
42   * you have to fall back to Java's ProcessBuilder.)
43   *
44   * Note that the first "arg" passed to an instance of this class is actually not an argument, but
45   * the name of the program or script to be executed. I accept this inconsistency because it simplifies
46   * the api and allows for shorter method names.
47   */
48  public class Launcher {
49      private final ProcessBuilder builder;
50      private String encoding;
51  
52      public Launcher(String... args) {
53          this.builder = new ProcessBuilder();
54          arg(args);
55      }
56  
57      public Launcher(FileNode dir, String... args) {
58          this(args);
59          dir(dir);
60      }
61  
62      //-- configuration
63  
64      public Launcher env(String key, String value) {
65          builder.environment().put(key, value);
66          return this;
67      }
68  
69      public Launcher arg(String... args) {
70          for (String a : args) {
71              builder.command().add(a);
72          }
73          return this;
74      }
75  
76      public Launcher args(List<String> args) {
77          builder.command().addAll(args);
78          return this;
79      }
80  
81      /** initializes the directory to execute the command in */
82      public Launcher dir(FileNode dir) {
83          return dir(dir.toPath().toFile(), dir.getWorld().getSettings().encoding);
84      }
85  
86      /** You'll normally use the dir(FileNode) method instead. */
87      public Launcher dir(File dir, String encoding) {
88          this.builder.directory(dir);
89          this.encoding = encoding;
90          return this;
91      }
92  
93      //-- execution
94  
95      public void execNoOutput() throws Failure {
96          String result;
97  
98          result = exec();
99          if (result.trim().length() > 0) {
100             throw new Failure(this, builder.command().get(0) + ": unexpected output " + result);
101         }
102     }
103 
104     public String exec() throws Failure {
105         StringWriter result;
106 
107         result = new StringWriter();
108         exec(result);
109         return result.getBuffer().toString();
110     }
111 
112     public void exec(Writer all) throws Failure {
113         exec(all, null);
114     }
115 
116     public void exec(Writer stdout, Writer stderr) throws Failure {
117         exec(stdout, stderr, true, null, true);
118     }
119 
120     /**
121      * Executes a command in this directory, wired with the specified streams. None of the argument stream is closed.
122      *
123      * @param stdout never null
124      * @param stderr may be null (which will redirect the error stream to stdout.
125      * @param flushDest true to flush stdout/stderr after every chunk read from the process
126      * @param stdin Has to be null when stdinInherit is true. Otherwise: may be null, which starts a process without input
127      */
128     public void exec(Writer stdout, Writer stderr, boolean flushDest, Reader stdin, boolean stdinInherit) throws Failure {
129         Process process;
130         int exit;
131         String output;
132         Pumper psout;
133         Pumper pserr;
134         Pumper psin;
135 
136         if (builder.directory() == null) {
137             // builder.start() does not check, I would not detect the problem until process.waitFor is called
138             // - that's to late because buffer would also be null
139             throw new IllegalStateException("Missing directory. Call dir() before invoking this method");
140         }
141         builder.redirectErrorStream(stderr == null);
142         if (stdinInherit) {
143             builder.redirectInput(ProcessBuilder.Redirect.INHERIT);
144             if (stdin != null) {
145                 throw new IllegalArgumentException();
146             }
147         }
148         try {
149             process = builder.start();
150         } catch (IOException e) {
151             throw new Failure(this, e);
152         }
153         psout = Pumper.create(process.getInputStream(), stdout, flushDest, false, encoding);
154         psout.start();
155         if (stderr != null) {
156             pserr = Pumper.create(process.getErrorStream(), stderr, flushDest, false, encoding);
157             pserr.start();
158         } else {
159             pserr = null;
160         }
161         if (stdin != null && !stdinInherit) {
162             psin = Pumper.create(stdin, process.getOutputStream(), true, true, encoding);
163             psin.start();
164         } else {
165             psin = null;
166         }
167         psout.finish(this);
168         if (pserr != null) {
169             pserr.finish(this);
170         }
171         try {
172             exit = process.waitFor();
173         } catch (InterruptedException e) {
174             throw new Interrupted(e);
175         }
176         if (psin != null) {
177             psin.finish(this);
178         }
179         if (exit != 0) {
180             if (stderr == null && stdout instanceof StringWriter) {
181                 output = ((StringWriter) stdout).getBuffer().toString();
182             } else {
183                 output = "";
184             }
185             throw new ExitCode(this, exit, output);
186         }
187     }
188 
189     //--
190 
191     /** If you need access to the internals. Most applications won't need this method. */
192     public ProcessBuilder getBuilder() {
193         return builder;
194     }
195 
196     @Override
197     public String toString() {
198         return "[" + builder.directory() + "] " + Separator.SPACE.join(builder.command());
199     }
200 }
