16  package net.oneandone.sushi.metadata.properties;
17  
18  import net.oneandone.sushi.metadata.Cardinality;
19  import net.oneandone.sushi.metadata.ComplexType;
20  import net.oneandone.sushi.metadata.Item;
21  import net.oneandone.sushi.metadata.SimpleType;
22  import net.oneandone.sushi.metadata.SimpleTypeException;
23  import net.oneandone.sushi.metadata.Type;
24  import net.oneandone.sushi.util.Separator;
25  
26  import java.util.ArrayList;
27  import java.util.Collection;
28  import java.util.Collections;
29  import java.util.List;
30  import java.util.Properties;
31  
32  /**
33   * Helper class to read properties. You'll usually not use this class directly, use Instance.toProperties instead.
34   */
35  public class Loader {
36      public static Object run(Properties src, Type type) throws LoadException {
37          return run(src, type, "");
38      }
39  
40      public static Object run(Properties src, Type type, String name) throws LoadException {
41          Loader loader;
42          Object result;
43  
44          loader = new Loader(src);
45          result = loader.load(IO.toExternal(name), type);
46          if (!src.isEmpty()) {
47              loader.error("unused properties: " + src.keySet());
48          }
49          if (!loader.errors.isEmpty()) {
50              throw new LoadException(result, Separator.RAW_LINE.join(loader.errors));
51          }
52          return result;
53      }
54  
55      private final Properties src;
56      private final List<String> errors;
57  
58      private Loader(Properties src) {
59          this.src = src;
60          this.errors = new ArrayList<>();
61      }
62  
63      private Object load(String key, Type type) {
64          ComplexType parent;
65          String value;
66          Object obj;
67          String childKey;
68          Cardinality card;
69          Collection<?> col;
70          
71          value = eat(key);
72          if (type instanceof SimpleType) {
73              if (value == null) {
74                  error(key + ": value not found");
75                  return type.newInstance();
76              }
77              try {
78                  return ((SimpleType) type).stringToValue(value);
79              } catch (SimpleTypeException e) {
80                  error(key + ": invalid value '" + value + "': " + e.getMessage());
81                  return type.newInstance();
82              }
83          } else {
84              if (value != null) {
85                  try {
86                      type = type.getSchema().type(Class.forName(value));
87                  } catch (ClassNotFoundException e) {
88                      error(key + ": class not found: " + value);
89                      return type.newInstance();
90                  }
91              } else {
92                  // type as specified in schema
93              }
94              parent = (ComplexType) type;
95              obj = parent.newInstance();
96              for (Item item : parent.items()) {
97                  childKey = join(key, IO.toExternal(item.getName()));
98                  card = item.getCardinality();
99                  if (item.getCardinality() == Cardinality.SEQUENCE) {
100                     col = loadIndexed(childKey, item.getType());
101                 } else {
102                     col = loadNormal(childKey, item.getType());
103                 }
104                 if (col.size() < card.min) {
105                     error(childKey + ": missing values: expected " + card.min + ", got " + col.size());
106                 } else if (col.size() > card.max) {
107                     error(childKey + ": to many values: expected " + card.max + ", got " + col.size());
108                 } else {
109                     item.set(obj, col);
110                 }
111             }
112             return obj;
113         }
114     }
115 
116     private void error(String message) {
117         errors.add(message);
118     }
119 
120     private Collection<?> loadIndexed(String key, Type type) {
121         List<Object> col;
122         String childKey;
123         
124         col = new ArrayList<>();
125         for (int i = 1; true; i++) {
126             childKey = key + "." + Integer.toString(i);
127             if (!contains(childKey)) {
128                 return col;
129             }
130             col.add(load(childKey, type));
131         }
132     }
133 
134     private Collection<?> loadNormal(String key, Type type) {
135         if (contains(key)) {
136             return Collections.singleton(load(key, type));
137         } else {
138             return Collections.EMPTY_LIST;
139         }
140     }
141     
142     private String eat(String key) {
143         return (String) src.remove(key);
144     }
145 
146     private boolean contains(String prefix) {
147         String prefixSlash;
148         String str;
149 
150         prefixSlash = prefix + "/";
151         // TODO: expensive
152         for (Object key : src.keySet()) {
153             str = (String) key;
154             if (str.equals(prefix)) {
155                 return true;
156             }
157             if (str.startsWith(prefixSlash)) {
158                 return true;
159             }
160         }
161         return false;
162     }
163 
164     private static String join(String first, String second) {
165         return first.length() == 0 ? second : first + "/" + second;
166     }
167 }
