16  package net.oneandone.sushi.metadata.xml;
17  
18  import net.oneandone.sushi.metadata.ComplexType;
19  import net.oneandone.sushi.metadata.Item;
20  import net.oneandone.sushi.metadata.ItemException;
21  import net.oneandone.sushi.metadata.Type;
22  import net.oneandone.sushi.metadata.Variable;
23  import org.xml.sax.Locator;
24  import org.xml.sax.SAXException;
25  
26  import java.util.ArrayList;
27  import java.util.HashMap;
28  import java.util.List;
29  import java.util.Map;
30  
31  public class ComplexElement extends Element {
32      private final ComplexType type;
33      private final Map<Item<?>, List<Object>> children;
34  
35      public ComplexElement(Item<?> owner, ComplexType type) {
36          super(owner);
37          
38          if (owner != null && !owner.getType().getType().isAssignableFrom(type.getType())) {
39              throw new IllegalArgumentException();
40          }
41          this.type = type;
42          this.children = new HashMap<Item<?>, List<Object>>();
43          for (Item<?> item : type.items()) {
44              children.put(item, new ArrayList<Object>());
45          }
46      }
47  
48      @Override
49      public Type getType() {
50          return type;
51      }
52      
53      @Override
54      public void addChild(Item<?> item, Object obj) {
55          children.get(item).add(obj);
56      }
57      
58      @Override
59      public boolean isEmpty() {
60          for (List<Object> values : children.values()) {
61              if (values.size() > 0) {
62                  return false;
63              }
64          }
65          return true;
66      }
67      
68      @Override
69      public Object create(List<SAXException> exceptions, Locator locator) {
70          Object object;
71          Item item;
72          
73          object = type.newInstance();
74          for (Map.Entry<Item<?>, List<Object>> entry : children.entrySet()) {
75              item = entry.getKey();
76              try {
77                  item.set(object, entry.getValue());
78              } catch (ItemException e) {
79                  exceptions.add(new SAXVariableException(new Variable<Object>(object, item), locator, e));
80              }            
81          }
82          return object;
83      }
84  
85      @Override
86      public boolean addContent(char[] ch, int ofs, int len) {
87          int max;
88          
89          max = ofs + len;
90          for (int i = ofs; i < max; i++) {
91              if (!Character.isWhitespace(ch[i])) {
92                  return false;
93              }
94          }
95          return true;
96      }
97  
98      @Override
99      public Item lookup(String child) {
100         return type.lookupXml(child);
101     }
102 }
