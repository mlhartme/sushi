16  package net.oneandone.sushi.metadata.xml;
17  
18  import net.oneandone.sushi.io.OS;
19  
20  import java.io.IOException;
21  import java.io.Writer;
22  
23  public class WriterTree extends Tree {
24      private final Writer dest;
25      private int indent;
26      private boolean strict;
27      private final String lineSeparator;
28  
29      public WriterTree(Writer dest, boolean strict) {
30      	this(dest, strict, OS.CURRENT.lineSeparator.getSeparator());
31      }
32  
33      public WriterTree(Writer dest, boolean strict, String lineSeparator) {
34          this.dest = dest;
35          this.indent = 0;
36          this.strict = strict;
37          this.lineSeparator = lineSeparator;
38      }
39  
40      @Override
41      public Writer done() throws IOException {
42          if (indent != 0) {
43              throw new IllegalStateException("" + indent);
44          }
45          dest.flush();
46          return dest;
47      }
48  
49      @Override
50      public void ref(String name, int id) throws IOException {
51          indent();
52          dest.write("<");
53          dest.write(name);
54          dest.write(" idref='");
55          dest.write(Integer.toString(id));
56          dest.write("'/>");
57          dest.write(lineSeparator);
58      }
59  
60      @Override
61      public void begin(String name, int id, String type, boolean withEnd) throws IOException {
62          indent();
63          dest.write("<");
64          dest.write(name);
65          if (id != -1) {
66              dest.write(" id='");
67              dest.write(Integer.toString(id));
68              dest.write('\'');
69          }
70          type(type);
71          if (withEnd) {
72              dest.write("/>");
73          } else {
74              indent++;
75              dest.write(">");
76          }
77          dest.write(lineSeparator);
78      }
79  
80      @Override
81      public void end(String name) throws IOException {
82          --indent;
83          indent();
84          dest.write("</");
85          dest.write(name);
86          dest.write('>');
87          dest.write(lineSeparator);
88      }
89  
90      @Override
91      public void text(String name, String typeAttribute, String text) throws IOException {
92          indent();
93          dest.write('<');
94          dest.write(name);
95          type(typeAttribute);
96          dest.write('>');
97          dest.write(net.oneandone.sushi.xml.Serializer.escapeEntities(text, strict));
98          dest.write("</");
99          dest.write(name);
100         dest.write('>');
101         dest.write(lineSeparator);
102     }
103 
104     private void type(String type) throws IOException {
105         if (type != null) {
106             dest.write(" type='");
107             dest.write(type);
108             dest.write('\'');
109         }
110     }
111 
112     private void indent() throws IOException {
113         for (int i = 0; i < indent; i++) {
114             dest.write("  ");
115         }
116     }
117 }
