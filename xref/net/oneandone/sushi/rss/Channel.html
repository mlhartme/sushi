16  package net.oneandone.sushi.rss;
17  
18  import net.oneandone.sushi.xml.Builder;
19  import net.oneandone.sushi.xml.Selector;
20  import net.oneandone.sushi.xml.XmlException;
21  import org.w3c.dom.Element;
22  
23  import java.util.ArrayList;
24  import java.util.List;
25  
26  public class Channel {
27      public static Channel fromXml(Selector selector, Element channel) throws XmlException {
28          Channel result;
29          
30          result = new Channel();
31          result.title = selector.string(channel, "title");
32          result.link = selector.string(channel, "link");
33          result.description = selector.string(channel, "description");
34          for (Element item : selector.elements(channel, "item")) {
35              result.items.add(Item.fromXml(selector, item));
36          }
37          return result;
38      }
39  
40      private String title;
41      private String link;
42      private String description;
43      private final List<Item> items;
44      
45      public Channel() {
46          this("", "", "");
47      }
48      
49      public Channel(String title, String link, String description) {
50          this.title = title;
51          this.link = link;
52          this.description = description;
53          this.items = new ArrayList<Item>();
54      }
55  
56      public String getTitle() {
57          return title;
58      }
59      public void setTitle(String title) {
60          this.title = title;
61      }
62      public String getLink() {
63          return link;
64      }
65      public void setLink(String link) {
66          this.link = link;
67      }
68      public String getDescription() {
69          return description;
70      }
71      public void setDescription(String description) {
72          this.description = description;
73      }
74      
75      public List<Item> items() {
76          return items;
77      }
78      
79      public void add(Item item, int maxItems) {
80          int remove;
81  
82          items.add(0, item);
83          remove = items.size() - maxItems;
84          while (remove-- > 0) {
85              items.remove(maxItems);
86          }
87      }
88      
89      public void addXml(Element rss) {
90          Element channel;
91          
92          channel = Builder.element(rss, "channel");
93          Builder.textElement(channel, "title", title);
94          Builder.textElement(channel, "link", link);
95          Builder.textElement(channel, "description", description);
96          for (Item item : items) {
97              item.addXml(channel);
98          }
99      }
100 }
