27  public class Base64 {
28      /** true create base64 */
29      private final boolean encoder;
30      private final byte[] srcBuffer;
31      private final byte[] destBuffer;
32      
33      public Base64(boolean encoder) {
34          this(encoder, 1024);
35      }
36  
37      public Base64(boolean encoder, int bufferSizeBase) {
38          int bufferSize;
39          
40          if (bufferSizeBase < 1) {
41              throw new IllegalArgumentException("" + bufferSizeBase);
42          }
43          bufferSize = bufferSizeBase * TRIPPLET_BYTES;
44          this.encoder = encoder;
45          if (encoder) {
46              this.srcBuffer = new byte[bufferSize];
47              this.destBuffer = new byte[toInt(encodedLength(bufferSize))];
48          } else {
49              this.srcBuffer = new byte[toInt(encodedLength(bufferSize))];
50              this.destBuffer = new byte[bufferSize];
51          }
52      }
53  
54      private static int toInt(long l) {
55          int i;
56          
57          i = (int) l;
58          if (i != l) {
59              throw new IllegalArgumentException("" + l);
60          }
61          return i;
62      }

64      public String run(String str) {
65          try {
66              return run(str, Settings.UTF_8);
67          } catch (UnsupportedEncodingException e) {
68              throw new IllegalStateException(e);
69          }
70      }
71  
72      public String run(String str, String encoding) throws UnsupportedEncodingException {
73          return new String(run(str.getBytes(encoding)), encoding);
74      }
75      
76      public byte[] run(byte ... bytes) {
77          return run(bytes, 0, bytes.length);
78      }
79      
80      public byte[] run(byte[] bytes, int ofs, int len) {
81          ByteArrayOutputStream dest;
82          
83          dest = new ByteArrayOutputStream();
84          try {
85              run(new ByteArrayInputStream(bytes, ofs, len), dest);
86          } catch (IOException e) {
87              throw new IllegalStateException(e);
88          }
89          return dest.toByteArray();
90      }
91      
92      public long run(InputStream src, OutputStream dest) throws IOException {
93          return run(src, dest, Long.MAX_VALUE);
94      }
95      
96      /** @return number of bytes actually written */
97      public long run(InputStream src, OutputStream dest, long maxSrcLength) throws IOException {
98          int srcCount;
99          int destCount;
100         long destLength; // toal length
101         long remaining;
102         
103         destLength = 0;
104         remaining = maxSrcLength;
105         while (true) {
106             srcCount = read(src, srcBuffer, (int) Math.min(srcBuffer.length, remaining));
107             if (srcCount == 0) {
108                 break;
109             }
110             destCount = encoder? encode(srcBuffer, srcCount, destBuffer) : decode(srcBuffer, srcCount, destBuffer);
111             dest.write(destBuffer, 0, destCount);
112             destLength += destCount;
113             remaining -= srcCount;
114         }
115         return destLength;
116     }

120     /**
121      * Make sure the buffer is filled as much as possible, even if the input stream
122      * delivers smaller chunks
123      * @return number of bytes actually read
124      */
125     public static int read(InputStream src, byte[] dest, int len) throws IOException {
126         int ofs;
127         int count;
128         
129         for (ofs = 0; ofs < len; ofs += count) {
130             count = src.read(dest, ofs, len - ofs);
131             if (count <= 0) {
132                 break;
133             }
134         }
135         return ofs;
136     }

140     public static long encodedLength(long decodedLength) {
141         long bits = decodedLength * EIGHTBIT;
142         long triplets = bits / TWENTYFOURBITGROUP;
143 
144         if (bits % TWENTYFOURBITGROUP != 0) {
145             return (triplets + 1) * 4;
146         } else {
147             return triplets * 4;
148         }
149     }

157     /**
158      * The base length.
159      */
160     private static final int BASELENGTH = 255;
161 
162     /**
163      * Lookup length.
164      */
165     private static final int LOOKUPLENGTH = 64;
166 
167     /**
168      * Used to calculate the number of bits in a byte.
169      */
170     private static final int EIGHTBIT = 8;
171 
172     /**
173      * Used when encoding something which has fewer than 24 bits.
174      */
175     private static final int SIXTEENBIT = 16;
176 
177     /**
178      * Used to determine how many bits data contains.
179      */
180     private static final int TWENTYFOURBITGROUP = 24;
181 
182 
183     /**
184      * Used to test the sign of a byte.
185      */
186     private static final int SIGN = -128;
187     
188     /**
189      * Byte used to pad output.
190      */
191     private static final byte PAD = (byte) '=';
192     
193     private static final int TRIPPLET_BYTES = TWENTYFOURBITGROUP / EIGHTBIT;
194     
195     // Create arrays to hold the base64 characters and a 
196     // lookup for base64 chars
197     private static final byte[] base64Alphabet = new byte[BASELENGTH];
198     private static final byte[] lookUpBase64Alphabet = new byte[LOOKUPLENGTH];
199 
200     // Populating the lookup and character arrays
201     static {
202         for (int i = 0; i < BASELENGTH; i++) {
203             base64Alphabet[i] = (byte) -1;
204         }
205         for (int i = 'Z'; i >= 'A'; i--) {
206             base64Alphabet[i] = (byte) (i - 'A');
207         }
208         for (int i = 'z'; i >= 'a'; i--) {
209             base64Alphabet[i] = (byte) (i - 'a' + 26);
210         }
211         for (int i = '9'; i >= '0'; i--) {
212             base64Alphabet[i] = (byte) (i - '0' + 52);
213         }
214 
215         base64Alphabet['+'] = 62;
216         base64Alphabet['/'] = 63;
217 
218         for (int i = 0; i <= 25; i++) {
219             lookUpBase64Alphabet[i] = (byte) ('A' + i);
220         }
221 
222         for (int i = 26, j = 0; i <= 51; i++, j++) {
223             lookUpBase64Alphabet[i] = (byte) ('a' + j);
224         }
225 
226         for (int i = 52, j = 0; i <= 61; i++, j++) {
227             lookUpBase64Alphabet[i] = (byte) ('0' + j);
228         }
229 
230         lookUpBase64Alphabet[62] = (byte) '+';
231         lookUpBase64Alphabet[63] = (byte) '/';
232     }
233 
234     private static boolean isBase64(byte octect) {
235         return octect == PAD || base64Alphabet[octect] != -1;
236     }

238     /**
239      * Encodes binary data using the base64 algorithm, no chunking.
240      *
241      * @param binaryData Array containing binary data to encode.
242      * @param binaryDataLength  number of bytes in binaryData.
243      * @param encodedData Array receiving encoded data.
244      * @return encoded data length
245      */
246     public static int encode(byte[] binaryData, int binaryDataLength, byte[] encodedData) {
247         int lengthDataBits = binaryDataLength * EIGHTBIT;
248         int fewerThan24bits = lengthDataBits % TWENTYFOURBITGROUP;
249         int numberTriplets = lengthDataBits / TWENTYFOURBITGROUP;
250         int encodedDataLength;
251 
252         if (fewerThan24bits != 0) {
253             //data not divisible by 24 bit
254             encodedDataLength = (numberTriplets + 1) * 4;
255         } else {
256             // 16 or 8 bit
257             encodedDataLength = numberTriplets * 4;
258         }
259         if (encodedDataLength > encodedData.length) {
260             throw new IllegalArgumentException(encodedDataLength + ">" + encodedData.length);
261         }
262 
263         byte k = 0, l = 0, b1 = 0, b2 = 0, b3 = 0;
264 
265         int encodedIndex = 0;
266         int dataIndex = 0;
267         int i;
268 
269         for (i = 0; i < numberTriplets; i++) {
270             dataIndex = i * 3;
271             b1 = binaryData[dataIndex];
272             b2 = binaryData[dataIndex + 1];
273             b3 = binaryData[dataIndex + 2];
274 
275             l = (byte) (b2 & 0x0f);
276             k = (byte) (b1 & 0x03);
277 
278             byte val1 =
279                 ((b1 & SIGN) == 0) ? (byte) (b1 >> 2) : (byte) ((b1) >> 2 ^ 0xc0);
280             byte val2 =
281                 ((b2 & SIGN) == 0) ? (byte) (b2 >> 4) : (byte) ((b2) >> 4 ^ 0xf0);
282             byte val3 =
283                 ((b3 & SIGN) == 0) ? (byte) (b3 >> 6) : (byte) ((b3) >> 6 ^ 0xfc);
284 
285             encodedData[encodedIndex] = lookUpBase64Alphabet[val1];
286             encodedData[encodedIndex + 1] = lookUpBase64Alphabet[val2 | (k << 4)];
287             encodedData[encodedIndex + 2] = lookUpBase64Alphabet[(l << 2) | val3];
288             encodedData[encodedIndex + 3] = lookUpBase64Alphabet[b3 & 0x3f];
289             encodedIndex += 4;
290         }
291 
292         // form integral number of 6-bit groups
293         dataIndex = i * 3;
294 
295         if (fewerThan24bits == EIGHTBIT) {
296             b1 = binaryData[dataIndex];
297             k = (byte) (b1 & 0x03);
298             byte val1 =
299                 ((b1 & SIGN) == 0) ? (byte) (b1 >> 2) : (byte) ((b1) >> 2 ^ 0xc0);
300             encodedData[encodedIndex] = lookUpBase64Alphabet[val1];
301             encodedData[encodedIndex + 1] = lookUpBase64Alphabet[k << 4];
302             encodedData[encodedIndex + 2] = PAD;
303             encodedData[encodedIndex + 3] = PAD;
304         } else if (fewerThan24bits == SIXTEENBIT) {
305 
306             b1 = binaryData[dataIndex];
307             b2 = binaryData[dataIndex + 1];
308             l = (byte) (b2 & 0x0f);
309             k = (byte) (b1 & 0x03);
310 
311             byte val1 = ((b1 & SIGN) == 0) ? (byte) (b1 >> 2) : (byte) ((b1) >> 2 ^ 0xc0);
312             byte val2 = ((b2 & SIGN) == 0) ? (byte) (b2 >> 4) : (byte) ((b2) >> 4 ^ 0xf0);
313             encodedData[encodedIndex] = lookUpBase64Alphabet[val1];
314             encodedData[encodedIndex + 1] = lookUpBase64Alphabet[val2 | (k << 4)];
315             encodedData[encodedIndex + 2] = lookUpBase64Alphabet[l << 2];
316             encodedData[encodedIndex + 3] = PAD;
317         }
318 
319         return encodedDataLength;
320     }

322     /**
323      * Decodes Base64 data into octects
324      *
325      * @param encoded Byte array containing Base64 data
326      * @param encodedLength number of bytes in base64Data
327      * @param binary Array containing decoded data.
328      * @return binary data length
329      */
330     public static int decode(byte[] encoded, int encodedLength, byte[] binary) {
331         byte b1, b2, b3, b4, marker0, marker1;
332         int binaryIndex;
333         int encodedIndex;
334         
335         encodedLength = discardNonBase64(encoded, encodedLength);
336         encodedIndex = 0;
337         binaryIndex = 0;
338         while (encodedIndex < encodedLength) {
339             b1 = base64Alphabet[encoded[encodedIndex++]];
340             b2 = base64Alphabet[encoded[encodedIndex++]];
341             marker0 = encoded[encodedIndex++];
342             marker1 = encoded[encodedIndex++];
343             if (marker0 == PAD) {
344                 // two PAD e.g. 3c[Pad][Pad]
345                 if (marker1 != PAD) {
346                     throw new IllegalArgumentException();
347                 }
348                 binary[binaryIndex] = (byte) (b1 << 2 | b2 >> 4);
349                 binaryIndex += 1;
350                 if (encodedIndex != encodedLength) {
351                     throw new IllegalStateException();
352                 }
353             } else if (marker1 == PAD) {
354                 // one PAD e.g. 3cQ[Pad]
355                 b3 = base64Alphabet[marker0];
356             
357                 binary[binaryIndex] = (byte) (b1 << 2 | b2 >> 4);
358                 binary[binaryIndex + 1] =
359                 (byte) (((b2 & 0xf) << 4) | ((b3 >> 2) & 0xf));
360                 binaryIndex += 2;
361                 if (encodedIndex != encodedLength) {
362                     throw new IllegalStateException();
363                 }
364             } else {
365                 // no PAD e.g 3cQl
366                 b3 = base64Alphabet[marker0];
367                 b4 = base64Alphabet[marker1];
368                 
369                 binary[binaryIndex] = (byte) (b1 << 2 | b2 >> 4);
370                 binary[binaryIndex + 1] = (byte) (((b2 & 0xf) << 4) | ((b3 >> 2) & 0xf));
371                 binary[binaryIndex + 2] = (byte) (b3 << 6 | b4);
372                 binaryIndex += 3;
373             } 
374         }
375         return binaryIndex;
376     }

378     /**
379      * Discards any characters outside of the base64 alphabet, per
380      * the requirements on page 25 of RFC 2045 - "Any characters
381      * outside of the base64 alphabet are to be ignored in base64
382      * encoded data."
383      *
384      * @param data The base-64 encoded data to groom
385      * @return number of bytes in groomed data.
386      */
387     static int discardNonBase64(byte[] data, int length) {
388         int dest = 0;
389         for (int i = 0; i < length; i++) {
390             if (isBase64(data[i])) {
391                 data[dest++] = data[i];
392             }
393         }
394         return dest;
395     }
