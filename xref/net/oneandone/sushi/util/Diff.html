22   * Output in unified format. See http://en.wikipedia.org/wiki/Diff

24  public class Diff {
25      public static String diff(String leftStr, String rightStr) {
26          return diff(leftStr, rightStr, false, 0);
27      }

29      public static String diff(String leftStr, String rightStr, boolean range, int context) {
30          return diff(leftStr, rightStr, range, context, false);
31      }

33      public static String diff(String leftStr, String rightStr, boolean range, int context, boolean escape) {
34          return diff(Separator.RAW_LINE.split(leftStr), Separator.RAW_LINE.split(rightStr), range, context, escape);
35      }

37      public static String diff(List<String> left, List<String> right, boolean range, int context, boolean escape) {
38          List<String> commons;
39          List<Chunk> chunks;
40          Chunk chunk;
41          StringBuilder result;
42          int ci;
43          Chunk last;
44  
45          commons = Lcs.compute(left, right);
46          chunks = diff(left, commons, right);
47          result = new StringBuilder();
48          last = null;
49          for (int c = 0; c < chunks.size(); c++) {
50              chunk = chunks.get(c);
51              if (range && (last == null || !last.touches(chunk, context))) {
52                  addRange(result, chunks, c, commons.size(), context);
53              }
54              ci = Math.max(chunk.common - context, c == 0 ? 0 : chunks.get(c - 1).common);
55              for (int i = ci; i < chunk.common; i++) {
56                  result.append(" ").append(commons.get(i));
57              }
58              if (chunk.delete > 0) {
59                  for (int i = chunk.left; i < chunk.left + chunk.delete; i++) {
60                      result.append('-');
61                      appendEscaped(left.get(i), escape, result);
62                  }
63              }
64              if (chunk.add.size() > 0) {
65                  for (String line : chunk.add) {
66                      result.append("+");
67                      appendEscaped(line, escape, result);
68                  }
69              }
70              ci = Math.min(chunk.common + context, c == chunks.size() - 1 ? commons.size() : chunks.get(c + 1).common - context);
71              for (int i = chunk.common; i < ci; i++) {
72                  result.append(" ").append(commons.get(i));
73              }
74              last = chunk;
75          }
76          return result.toString();
77      }

79      public static void appendEscaped(String str, boolean escape, StringBuilder dest) {
80          int max;
81          char c;
82          if (escape) {
83              max = str.length();
84              for (int i = 0; i < max; i++) {
85                  c = str.charAt(i);
86                  switch (c) {
87                  case '\t':
88                      dest.append("\\t");
89                      break;
90                  case '\r':
91                      dest.append("\\r");
92                      break;
93                  case '\n':
94                      dest.append("\\n");
95                      break;
96                  default:
97                      if (c < ' ') {
98                          dest.append('[').append((int) c).append(']');
99                      } else {
100                         dest.append(c);
101                     }
102                 }
103             }
104         } else {
105             dest.append(str);
106         }
107     }
108     public static List<Chunk> diff(List<String> left, List<String> commons, List<String> right) {
109         Chunk chunk;
110         List<Chunk> result;
111         int li;
112         int ri;
113         int lmax;
114         int rmax;
115         String common;
116 
117         result = new ArrayList<>();
118         lmax = left.size();
119         rmax = right.size();
120         li = 0;
121         ri = 0;
122         for (int ci = 0; ci <= commons.size(); ci++) {
123             common = ci < commons.size() ? commons.get(ci) : null;
124             if (li < lmax && !left.get(li).equals(common)) {
125                 chunk = new Chunk(li, ci, ri);
126                 result.add(chunk);
127                 do {
128                     li++;
129                 } while (li < lmax && !left.get(li).equals(common));
130                 chunk.delete = li - chunk.left;
131             } else {
132                 // definite assignment
133                 chunk = null;
134             }
135             if (ri < rmax && !right.get(ri).equals(common)) {
136                 if (chunk == null) {
137                     chunk = new Chunk(li, ci, ri);
138                     result.add(chunk);
139                 }
140                 do {
141                     chunk.add.add(right.get(ri++));
142                 } while (ri < rmax && !right.get(ri).equals(common));
143             }
144             li++;
145             ri++;
146         }
147         return result;
148     }

150     private static void addRange(StringBuilder result, List<Chunk> chunks, int ofs, int cmax, int context) {
151         Chunk first;
152         Chunk prev;
153         Chunk current;
154         int leftStart;
155         int leftCount;
156         int rightStart;
157         int rightCount;
158 
159         first = chunks.get(ofs);
160         leftStart = Math.max(first.left - context, 0);
161         rightStart = Math.max(first.right - context, 0);
162         leftCount = first.left - leftStart;
163         rightCount = first.right - rightStart;
164         prev = null;
165         for ( ; ofs < chunks.size(); ofs++) {
166             current = chunks.get(ofs);
167             if (prev != null && !prev.touches(current, context)) {
168                 break;
169             }
170             leftCount += current.delete;
171             rightCount += current.add.size();
172             prev = current;
173         }
174         current = chunks.get(ofs - 1);
175         leftCount += current.common - first.common;
176         leftCount += Math.min(current.common + context, cmax) - current.common;
177         rightCount += current.common - first.common;
178         rightCount += Math.min(current.common + context, cmax) - current.common;
179         result.append("@@ -");
180         addRange(result, leftStart, leftCount);
181         result.append(" +");
182         addRange(result, rightStart, rightCount);
183         result.append(" @@\n");
184     }
185 
186     private static void addRange(StringBuilder result, int idx, int count) {
187         result.append(idx + 1);
188         if (count != 1) {
189             result.append(',').append(count);
190         }
191     }
192 
193     private static class Chunk {
194         public final int left;
195         public final int common;
196         public final int right;
197         public int delete;
198         public final List<String> add;
199 
200         public Chunk(int left, int common, int right) {
201             this.left = left;
202             this.right = right;
203             this.common = common;
204             this.add = new ArrayList<String>();
205         }
206 
207         public boolean touches(Chunk chunk, int context) {
208             return Math.abs(common - chunk.common) <= context * 2;
209         }
210 
211         public String range() {
212             return "@@ -" + (left + 1) + "," + delete + " +" + (right + 1) + "," + add.size() + " @@\n";
213         }
214     }
