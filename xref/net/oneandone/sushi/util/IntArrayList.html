16  package net.oneandone.sushi.util;
17  
18  import java.io.Serializable;
19  
20  /**
21   * List of int elements. Similar to java.lang.List or
22   * java.util.ArrayList, but it stores primitive int values.
23   * Generic collections for primitive types whould remove the
24   * need for IntArrayList. I implemented only those methods
25   * that I acually need.
26   */
27  
28  public class IntArrayList implements IntCollection, Serializable {
29      /** Storage for elements. */
30      private int[] data;
31  
32      /** Number of data elements actually used. */
33      private int size;
34  
35      public IntArrayList(int size, int[] data) {
36          this.size = size;
37          this.data = data;
38      }
39  
40      public IntArrayList(int[] init) {
41          size = init.length;
42          data = new int[size];
43          System.arraycopy(init, 0, data, 0, size);
44      }
45      //--------------------------------------------------------------
46  
47      /** Creates a new empty List, initial size is 32. */
48      public IntArrayList() {
49          this(32);
50      }
51  
52      public IntArrayList(int initialSize) {
53          data = new int[initialSize];
54          size = 0;
55      }
56  
57      /**
58       * Copy constructor.
59       * @param  orig  List that supplies the initial elements for
60       *               the new List.
61       */
62      public IntArrayList(IntArrayList orig) {
63          data = new int[orig.data.length];
64          size = orig.size;
65          System.arraycopy(orig.data, 0, data, 0, size);
66      }
67  
68      //-----------------------------------------------------------------
69  
70      @Override
71      public int hashCode() {
72          return size();
73      }
74      
75      @Override
76      public boolean equals(Object obj) {
77          IntArrayList operand;
78          int i;
79  
80          if (obj instanceof IntArrayList) {
81              operand = (IntArrayList) obj;
82              if (size == operand.size) {
83                  for (i = 0; i < size; i++) {
84                      if (data[i] != operand.data[i]) {
85                          return false;
86                      }
87                  }
88                  return true;
89              }
90          }
91          return false;
92      }
93  
94      /**
95       * Gets an element from the List.
96       * @param   idx  index of the element asked for
97       * @return  selected element
98       */
99      public int get(int idx) {
100         return data[idx];
101     }
102 
103     /**
104      * Replaces an element in the List.
105      * @param  ele  new element
106      * @param  idx  index of the element to be replaced
107      */
108     public void set(int idx, int ele) {
109         data[idx] = ele;
110     }
111 
112     public void ensureCapacity(int min) {
113         int[] tmp;
114         int old;
115         int capacity;
116 
117         old = data.length;
118         if (min > old) {
119             tmp = data;
120             capacity = (old * 5) / 3 + 1;
121             if (capacity < min) {
122                 capacity = min;
123             }
124             data = new int[capacity];
125             System.arraycopy(tmp, 0, data, 0, size);
126         }
127     }
128 
129     /**
130      * Adds an element to the List. All following elements
131      * are moved up by one index.
132      * @param  idx  where to insert the new element
133      * @param  ele  new element
134      */
135     public void add(int idx, int ele) {
136         ensureCapacity(size + 1);
137         System.arraycopy(data, idx, data, idx + 1, size - idx);
138         data[idx] = ele;
139         size++;
140     }
141 
142     /**
143      * Adds an element to the end of the List.
144      * @param  ele  new element
145      */
146     public void add(int ele) {
147         ensureCapacity(size + 1);
148         data[size++] = ele;
149     }
150 
151     public void addAll(IntArrayList op) {
152         ensureCapacity(size + op.size);
153         System.arraycopy(op.data, 0, data, size, op.size);
154         size += op.size;
155     }
156 
157 
158     /**
159      * Removes an element from the List. All following elements
160      * are moved down by one index.
161      * @param  idx  index of the element to remove
162      */
163     public void remove(int idx) {
164         size--;
165         System.arraycopy(data, idx + 1, data, idx, size - idx);
166     }
167 
168     /**
169      * Removes all elements.
170      */
171     public void clear() {
172         size = 0;
173     }
174 
175     /**
176      * Searches an element.
177      * @param   ele  element to look for
178      * @return  index of the first element found; -1 if nothing was found
179      */
180     public int indexOf(int ele) {
181         int i;
182 
183         for (i = 0; i < size; i++) {
184             if (data[i] == ele) {
185                 return i;
186             }
187         }
188         return -1;
189     }
190 
191     public boolean contains(int ele) {
192         return indexOf(ele) != -1;
193     }
194 
195     /**
196      * Returns the number of elements in the List.
197      * @return number of elements
198      */
199     public int size() {
200         return size;
201     }
202 
203     public boolean isEmpty() {
204         return size == 0;
205     }
206 
207     /**
208      * Creates an array with all elements of the List.
209      * @return  the array requested
210      */
211     public int[] toArray() {
212         return toArray(new int[size]);
213     }
214 
215     public int[] toArray(int[] result) {
216         if (result.length < size) {
217             result = new int[size];
218         }
219         System.arraycopy(data, 0, result, 0, size);
220         return result;
221     }
222 
223     //-----------------------------------------------------------------
224 
225     /**
226      * Returns a string representation.
227      * @return string representation
228      */
229     @Override
230     public String toString() {
231         StringBuilder buffer;
232         int i, max;
233 
234         max = size();
235         buffer = new StringBuilder();
236         for (i = 0; i < max; i++) {
237             buffer.append(' ');
238             buffer.append(get(i));
239         }
240         return buffer.toString();
241     }
242 }
