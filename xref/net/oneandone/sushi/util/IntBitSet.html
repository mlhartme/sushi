16  package net.oneandone.sushi.util;
17  
18  import java.io.Serializable;
19  import java.util.List;
20  
21  /**
22   * <p>A set of non-negative int values. IntBitSet is particularly usefull
23   * to store array indexes. IntBitSet is similar to BitSet, but with
24   * additional functionality, mainly subset tests and enumeration of elements.
25   * Method names follow the style of collections rather than BitSet
26   * because usually collections are better known. </p>
27   *
28   * <p>Tuned for speed. Each int value is represented by a staticallly
29   * chosen bit in a data array. Thus the space required for a set
30   * determined by largest element. This is memory-consuming. However.
31   * if the set stores array indexes and memory is available for a big
32   * array, there is probably enough memory to store big indexes as well.
33   * </p>
34   *
35   * <p>The method names follow java.util.BitSet instead of java.util.Set.
36   * The main reasons are: 1. IntBitSet stores primitive values instead of
37   * Objects. 2. IntBitSet is to limited to be called a collection, because
38   * adding large int values result in a huge memory consumption. </p>
39   *
40   * <p>Java specifies that the right argument of shift operations is
41   * implicitly bit-anded with 0x1f. I rely on this and don't mask any
42   * indexes. </p>
43   */
44  
45  public class IntBitSet implements IntCollection, Serializable {
46      public static IntBitSet with(int ... elements) {
47          IntBitSet result;
48  
49          result = new IntBitSet();
50          for (int e : elements) {
51              result.add(e);
52          }
53          return result;
54      }
55  
56      /** Number to shift in order to switch from an index to an
57          element. */
58      private static final int SHIFT = 5;
59  
60      /** Bit mask to obtain SHIFT bits. */
61      private static final int MASK = 0x1f;
62  
63      /** Number of components data is grown if necessary. */
64      private static final int GROW = 8;
65  
66  
67      /** Storage for the elements. */
68      private int[] data;
69  
70      //-------------------------------------------------------------
71  
72      /** Creates an empty set. */
73      public IntBitSet() {
74          data = new int[GROW];
75      }
76  
77      /** Copy constructor. */
78      public IntBitSet(IntBitSet set) {
79          data = new int[set.data.length];
80          System.arraycopy(set.data, 0, data, 0, data.length);
81      }
82  
83      public IntBitSet(int[] data) {
84          this.data = data;
85      }

88      //-------------------------------------------------------------------
89      // iteration
90  
91      /**
92       * Gets the first element of the set.
93       * @return   first element of the set, -1 if the set is empty
94       */
95      public int first() {
96          int val;
97          int i;
98  
99          for (i = 0; i < data.length; i++) {
100             if (data[i] != 0) {
101                 val = data[i];
102                 i <<= SHIFT;
103                 while ((val & 1) == 0) {
104                     val >>>= 1;
105                     i++;
106                 }
107                 return i;
108             }
109         }
110         return -1;
111     }
112 
113     /**
114      * Gets the last element of the set.
115      * @return   last element of the set, -1 if the set is empty
116      */
117     public int last() {
118         int val;
119         int i;
120 
121         for (i = data.length - 1; i >= 0; i--) {
122             if (data[i] != 0) {
123                 val = data[i] >>> 1;
124                 i <<= SHIFT;
125                 while (val != 0) {
126                     val >>>= 1;
127                     i++;
128                 }
129                 return i;
130             }
131         }
132         return -1;
133     }
134 
135     /**
136      * Gets the element following ele.
137      * @param   ele  the element to start with
138      * @return  the element following ele, -1 if nothing follows
139      */
140     public int next(int ele) {
141         int idx, bit, val;
142 
143         idx = ele >> SHIFT;
144         bit = ele & MASK;
145         val = data[idx] >>> bit;
146 
147         do {
148             val >>>= 1;
149             bit++;
150             if (val == 0) {
151                 idx++;
152                 if (idx == data.length) {
153                     return -1;
154                 }
155                 val = data[idx];
156                 bit = 0;
157             }
158         } while ((val & 1) == 0);
159         return (idx << SHIFT) + bit;
160     }
161 
162     public int prev(int ele) {
163         // TODO: very expensive
164         while (ele > 0) {
165             ele--;
166             if (contains(ele)) {
167                 return ele;
168             }
169         }
170         return -1;
171     }

173     //-----------------------------------------------------------------------
174     // access elements
175 
176     /**
177      * Lookup a specific element.
178      * @param   ele  element to lookup
179      * @return  true, if ele is contained in the set
180      */
181     public boolean contains(int ele) {
182         int idx;
183 
184         idx = ele >> SHIFT;
185         if (idx >= data.length) {
186             return false;
187         }
188         return (data[idx] & (1 << ele)) != 0;
189     }
190 
191     public boolean containsSome(IntBitSet rest) {
192         int ele;
193 
194         for (ele = first(); ele != -1; ele = next(ele)) {
195             if (rest.contains(ele)) {
196                 return true;
197             }
198         }
199         return false;
200     }
201 
202     /**
203      * Add an element.
204      * @param  ele  the element to add
205      */
206     public void add(int ele) {
207         int idx;
208 
209         idx = ele >> SHIFT;
210         if (idx >= data.length) {
211             resize(idx + 1 + GROW);
212         }
213         data[ele >> SHIFT] |= (1 << ele);
214     }
215 
216     /**
217      * Add all elements in the indicated range. Nothing is set for first &gt; last.
218      * @param  first   first element to add
219      * @param  last    last element to add
220      */
221     public void addRange(int first, int last) {
222         int ele;
223 
224         for (ele = first; ele <= last; ele++) {
225             add(ele);
226         }
227     }
228 
229     /**
230      * Remove an element from the set.
231      * @param  ele  element to remove
232      */
233     public void remove(int ele) {
234         int idx;
235 
236         idx = ele >> SHIFT;
237         if (idx < data.length) {
238             data[idx] &= ~(1 << ele);
239         }
240     }
241 
242     /**
243      * Remove all elements in the indicated range.
244      * @param  first  first element to remove
245      * @param  last   last element to remove
246      */
247     public void removeRange(int first, int last) {
248         int ele;
249 
250         for (ele = first; ele <= last; ele++) {
251             remove(ele);
252         }
253     }
254 
255     /**
256      * Remove all elements.
257      */
258     public void clear() {
259         int i;
260 
261         for (i = 0; i < data.length; i++) {
262             data[i] = 0;
263         }
264     }

269     @Override

271         return size();

274     /**
275      * Comparison.
276      * @param   obj  the Object to compare with
277      * @return  true, if obj is a set and contains exactly the
278      *          same elements as this set
279      */
280     @Override
281     public boolean equals(Object obj) {
282         IntBitSet set;
283         int i;
284         int[] x;
285         int[] y;
286 
287         if (!(obj instanceof IntBitSet)) {
288             return false;
289         }
290 
291         set = (IntBitSet) obj;
292         if (data.length < set.data.length) {
293             x = data;
294             y = set.data;
295         } else {
296             x = set.data;
297             y = data;
298         }
299 
300         for (i = 0; i < x.length; i++) {
301             if (x[i] != y[i]) {
302                 return false;
303             }
304         }
305         for ( ; i < y.length; i++) {
306             if (y[i] != 0) {
307                 return false;
308             }
309         }
310 
311         return true;
312     }
313 
314     /**
315      * The subset test.
316      * @param   set  the set to be compared with
317      * @return  true, if the argument is contained
318      */
319     public boolean containsAll(IntBitSet set) {
320         int i, end;
321 
322         if (data.length < set.data.length) {
323             end = data.length;
324             for (i = end; i < set.data.length; i++) {
325                 if (set.data[i] != 0) {
326                     return false;
327                 }
328             }
329         } else {
330             end = set.data.length;
331         }
332 
333         for (i = 0; i < end; i++) {
334             // any bits in set where this has none?  -> false
335             if ((set.data[i] & ~data[i])!= 0) {
336                 return false;
337             }
338         }
339         return true;
340     }

345     /**
346      * Computes the intersection. The result is stored in this set.
347      * @param  set  the set to be combined with */
348     public void retainAll(IntBitSet set) {
349         int i;
350 
351         if (set.data.length > data.length) {
352             resize(set.data.length);
353         }
354         for (i = 0; i < set.data.length; i++) {
355             data[i] &= set.data[i];
356         }
357         for ( ; i < data.length; i++) {
358             data[i] = 0;
359         }
360     }
361 
362     /**
363      * Computes the set difference. The result is stored in this set.
364      * @param   set   the set to be combined with
365      */
366     public void removeAll(IntBitSet set) {
367         int i, end;
368 
369         end = (set.data.length < data.length)? set.data.length : data.length;
370         for (i = 0; i < end; i++) {
371             data[i] &= ~set.data[i];
372         }
373     }
374 
375     /**
376      * Computes the union. The result is stored in this set.
377      * @param   set   the set to be combined with
378      */
379     public void addAll(IntBitSet set) {
380         int i;
381 
382         if (set.data.length > data.length) {
383             resize(set.data.length);
384         }
385 
386         for (i = 0; i < set.data.length; i++) {
387             data[i] |= set.data[i];
388         }
389     }
390 
391     public void addAllSets(IntBitSet[] related) {
392         int ele;
393 
394         for (ele = first(); ele != -1; ele = next(ele)) {
395             addAll(related[ele]);
396         }
397     }

403     /**
404      * Counts the elements.
405      * @return  number of elements in this set
406      */
407     public int size() {
408         int ele, count;
409 
410         count = 0;
411         for (ele = first(); ele != -1; ele = next(ele)) {
412             count++;
413         }
414         return count;
415     }

417     /**
418      * Tests for the empty set.
419      * @return  true, if the set contains no elements
420      */
421     public boolean isEmpty() {
422         return first() == -1;
423     }
424 
425     /**
426      * Creates an array with all elements of the set.
427      * @return   array with all elements of the set
428      */
429     public int[] toArray() {
430         int i, ele;
431         int[] result;
432 
433         result = new int[size()];
434         for (ele = first(), i = 0; ele != -1; ele = next(ele), i++) {
435             result[i] = ele;
436         }
437         return result;
438     }
439 
440     /**
441      * Changes data to have data.length >= s components. All data
442      * components that fit into the new size are preserved.
443      * @param   s   new data size wanted
444      */
445     private void resize(int s) {
446         int[] n;
447         int count;
448 
449         n = new int[s];
450         count = (s < data.length)? s : data.length;
451         System.arraycopy(data, 0, n, 0, count);
452         data = n;
453     }

458     /**
459      * Returns a String representation of the set.
460      * @return  a String listing all elements
461      */
462     @Override
463     public String toString() {
464         StringBuilder buffer;
465         int ele;
466 
467         buffer = new StringBuilder("{");
468         for (ele = first(); ele != -1; ele = next(ele)) {
469             buffer.append(' ').append(ele);
470         }
471         buffer.append(" }");
472         return buffer.toString();
473     }
474 
475     /**
476      * Returns a String representation with each element
477      * represented by a String taken from a Symboltable.
478      * @param   symbols  supplies the symbols to represent the elements
479      *                   of the set; a RuntimeException is thrown if an
480      *                   element is not found here
481      * @return  a String listing all elements of the set
482      */
483     public String toString(List<?> symbols) {
484          StringBuilder buf;
485          int ele;
486          int max;
487 
488          max = symbols.size();
489          buf = new StringBuilder("{");
490          for (ele = first(); ele != -1; ele = next(ele)) {
491              buf.append(" ");
492              if (ele < max) {
493                  buf.append(symbols.get(ele));
494              } else {
495                  buf.append('<').append(ele).append('>');
496              }
497          }
498          buf.append(" }");
499          return buf.toString();
500     }
